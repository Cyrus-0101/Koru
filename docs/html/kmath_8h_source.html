1395KINLINE mat4 quat_to_mat4(quat q) {
1396 mat4 out_matrix = mat4_identity();
1397
1398 // https://stackoverflow.com/questions/1556260/convert-quaternion-rotation-to-rotation-matrix
1399
1400 quat n = quat_normalize(q);
1401
1402 out_matrix.data[0] = 1.0f - 2.0f * n.y * n.y - 2.0f * n.z * n.z;
1403 out_matrix.data[1] = 2.0f * n.x * n.y - 2.0f * n.z * n.w;
1404 out_matrix.data[2] = 2.0f * n.x * n.z + 2.0f * n.y * n.w;
1405
1406 out_matrix.data[4] = 2.0f * n.x * n.y + 2.0f * n.z * n.w;
1407 out_matrix.data[5] = 1.0f - 2.0f * n.x * n.x - 2.0f * n.z * n.z;
1408 out_matrix.data[6] = 2.0f * n.y * n.z - 2.0f * n.x * n.w;
1409
1410 out_matrix.data[8] = 2.0f * n.x * n.z - 2.0f * n.y * n.w;
1411 out_matrix.data[9] = 2.0f * n.y * n.z + 2.0f * n.x * n.w;
1412 out_matrix.data[10] = 1.0f - 2.0f * n.x * n.x - 2.0f * n.y * n.y;
1413
1414 return out_matrix;
1415}

1417// Calculates a rotation matrix based on the quaternion and the passed in center point.
1425KINLINE mat4 quat_to_rotation_matrix(quat q, vec3 center) {
1426 mat4 out_matrix;
1427
1428 f32* o = out_matrix.data;
1429 o[0] = (q.x * q.x) - (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1430 o[1] = 2.0f * ((q.x * q.y) + (q.z * q.w));
1431 o[2] = 2.0f * ((q.x * q.z) - (q.y * q.w));
1432 o[3] = center.x - center.x * o[0] - center.y * o[1] - center.z * o[2];
1433
1434 o[4] = 2.0f * ((q.x * q.y) - (q.z * q.w));
1435 o[5] = -(q.x * q.x) + (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1436 o[6] = 2.0f * ((q.y * q.z) + (q.x * q.w));
1437 o[7] = center.y - center.x * o[4] - center.y * o[5] - center.z * o[6];
1438
1439 o[8] = 2.0f * ((q.x * q.z) + (q.y * q.w));
1440 o[9] = 2.0f * ((q.y * q.z) - (q.x * q.w));
1441 o[10] = -(q.x * q.x) - (q.y * q.y) + (q.z * q.z) + (q.w * q.w);
1442 o[11] = center.z - center.x * o[8] - center.y * o[9] - center.z * o[10];
1443
1444 o[12] = 0.0f;
1445 o[13] = 0.0f;
1446 o[14] = 0.0f;
1447 o[15] = 1.0f;
1448 return out_matrix;
1449}

1479KINLINE quat quat_slerp(quat q_0, quat q_1, f32 percentage) {
1480 quat out_quaternion;
1481 // Source: https://en.wikipedia.org/wiki/Slerp
1482 // Only unit quaternions are valid rotations.
1483 // Normalize to avoid undefined behavior.
1484 quat v0 = quat_normalize(q_0);
1485 quat v1 = quat_normalize(q_1);
1486
1487 // Compute the cosine of the angle between the two vectors.
1488 f32 dot = quat_dot(v0, v1);
1489
1490 // If the dot product is negative, slerp won't take
1491 // the shorter path. Note that v1 and -v1 are equivalent when
1492 // the negation is applied to all four components. Fix by
1493 // reversing one quaternion.
1494 if (dot < 0.0f) {
1495 v1.x = -v1.x;
1496 v1.y = -v1.y;
1497 v1.z = -v1.z;
1498 v1.w = -v1.w;
1499 dot = -dot;
1500 }
1501
1502 const f32 DOT_THRESHOLD = 0.9995f;
1503 if (dot > DOT_THRESHOLD) {
1504 // If the inputs are too close for comfort, linearly interpolate
1505 // and normalize the result.
1506 out_quaternion = (quat){
1507 v0.x + ((v1.x - v0.x) * percentage),
1508 v0.y + ((v1.y - v0.y) * percentage),
1509 v0.z + ((v1.z - v0.z) * percentage),
1510 v0.w + ((v1.w - v0.w) * percentage)};
1511
1512 return quat_normalize(out_quaternion);
1513 }
1514
1515 // Since dot is in range [0, DOT_THRESHOLD], acos is safe
1516 f32 theta_0 = kacos(dot); // theta_0 = angle between input vectors
1517 f32 theta = theta_0 * percentage; // theta = angle between v0 and result
1518 f32 sin_theta = ksin(theta); // compute this value only once
1519 f32 sin_theta_0 = ksin(theta_0); // compute this value only once
1520
1521 f32 s0 = kcos(theta) - dot * sin_theta / sin_theta_0; // == sin(theta_0 - theta) / sin(theta_0)
1522 f32 s1 = sin_theta / sin_theta_0;
1523
1524 return (quat){
1525 (v0.x * s0) + (v1.x * s1),
1526 (v0.y * s0) + (v1.y * s1),
1527 (v0.z * s0) + (v1.z * s1),
1528 (v0.w * s0) + (v1.w * s1)};
1529}

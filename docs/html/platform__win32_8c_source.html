39// Initializes the platform: registers window class, creates a window, sets up the timer.
40b8 platform_startup(
41 platform_state *plat_state,
42 const char *application_name,
43 i32 x,
44 i32 y,
45 i32 width,
46 i32 height) {
47 // Allocate and set internal state
48 plat_state->internal_state = malloc(sizeof(internal_state));
49 internal_state *state = (internal_state *)plat_state->internal_state;
50 state->h_instance = GetModuleHandleA(0);
51
52 // Load default icon
53 HICON icon = LoadIcon(state->h_instance, IDI_APPLICATION);
54
55 // Setup and register window class.
56 WNDCLASSA wc;
57 memset(&wc, 0, sizeof(wc));
58 wc.style = CS_DBLCLKS; // Get double-clicks
59 wc.lpfnWndProc = win32_process_message;
60 wc.cbClsExtra = 0;
61 wc.cbWndExtra = 0;
62 wc.hInstance = state->h_instance;
63 wc.hIcon = icon;
64 wc.hCursor = LoadCursor(NULL, IDC_ARROW); // NULL; // Manage the cursor manually
65 wc.hbrBackground = NULL; // Transparent
66 wc.lpszClassName = "koru_window_class";
67
68 if (!RegisterClassA(&wc)) {
69 MessageBoxA(0, "Window registration failed", "Error", MB_ICONEXCLAMATION | MB_OK);
70 return FALSE;
71 }
72
73 // Desired client area dimensions
74 u32 client_x = x;
75 u32 client_y = y;
76 u32 client_width = width;
77 u32 client_height = height;
78
79 // These will be adjusted for window borders
80 u32 window_x = client_x;
81 u32 window_y = client_y;
82 u32 window_width = client_width;
83 u32 window_height = client_height;
84
85 // Window styles
86 u32 window_style = WS_OVERLAPPED | WS_SYSMENU | WS_CAPTION | WS_MAXIMIZEBOX | WS_MINIMIZEBOX | WS_THICKFRAME;
87 u32 window_ex_style = WS_EX_APPWINDOW;
88
89 // Adjust for window border to get correct client area size
90 RECT border_rect = {0, 0, 0, 0};
91 AdjustWindowRectEx(&border_rect, window_style, 0, window_ex_style);
92
93 // In this case, the border rectangle is negative.
94 window_x += border_rect.left;
95 window_y += border_rect.top;
96
97 // Grow by the size of the OS border.
98 window_width += border_rect.right - border_rect.left;
99 window_height += border_rect.bottom - border_rect.top;
100
101 HWND handle = CreateWindowExA(
102 window_ex_style, "koru_window_class", application_name,
103 window_style, window_x, window_y, window_width, window_height,
104 0, 0, state->h_instance, 0);
105
106 if (handle == 0) {
107 MessageBoxA(NULL, "Window creation failed!", "Error!", MB_ICONEXCLAMATION | MB_OK);
108
109 KFATAL("Window creation failed!");
110 return FALSE;
111 } else {
112 state->hwnd = handle;
113 }
114
115 // Show window (with input focus)
116 b32 should_activate = 1; // TODO: if the window should not accept input, this should be false.
117 i32 show_window_command_flags = should_activate ? SW_SHOW : SW_SHOWNOACTIVATE;
118 // If initially minimized, use SW_MINIMIZE : SW_SHOWMINNOACTIVE;
119 // If initially maximized, use SW_SHOWMAXIMIZED : SW_MAXIMIZE
120 ShowWindow(state->hwnd, show_window_command_flags);
121
122 // Setup high-performance timer
123 LARGE_INTEGER frequency;
124 QueryPerformanceFrequency(&frequency);
125 clock_frequency = 1.0 / (f64)frequency.QuadPart;
126 QueryPerformanceCounter(&start_time);
127
128 return TRUE;
129}

236LRESULT CALLBACK win32_process_message(HWND hwnd, u32 msg, WPARAM w_param, LPARAM l_param) {
237 switch (msg) {
238 case WM_ERASEBKGND:
239 // Notify the OS that erasing will be handled by the application to prevent flicker.
240 return 1;
241 case WM_CLOSE:
242 event_context data = {};
243 event_fire(EVENT_CODE_APPLICATION_QUIT, 0, data);
244 return TRUE;
245 case WM_DESTROY:
246 PostQuitMessage(0);
247 return 0;
248 case WM_SIZE: {
249 // Get the updated size.
250 RECT r;
251 GetClientRect(hwnd, &r);
252 u32 width = r.right - r.left;
253 u32 height = r.bottom - r.top;
254
255 /*
256 * Fire the event. The application layer should pick this up,
257 * but not handle it as it shouldn't be visible to other parts
258 * of the application.
259 */
260 event_context context;
261
262 context.data.u16[0] = (u16)width;
263 context.data.u16[1] = (u16)height;
264
265 event_fire(EVENT_CODE_RESIZED, 0, context);
266 } break;
267 case WM_KEYDOWN:
268 case WM_SYSKEYDOWN:
269 case WM_KEYUP:
270 case WM_SYSKEYUP: {
271 // Key pressed/released
272 b8 pressed = (msg == WM_KEYDOWN || msg == WM_SYSKEYDOWN);
273 keys key = (u16)w_param;
274
275 // Pass to the input subsystem for processing
276 input_process_key(key, pressed)
277
278 } break;
279 case WM_MOUSEMOVE: {
280 // Mouse move
281 i32 x_position = GET_X_LPARAM(l_param);
282 i32 y_position = GET_Y_LPARAM(l_param);
283
284 // Pass to the input subsystem for processing
285 input_process_mouse_move(x_position, y_position);
286 } break;
287 case WM_MOUSEWHEEL: {
288 i32 z_delta = GET_WHEEL_DELTA_WPARAM(w_param);
289
290 if (z_delta != 0) {
291 // Flatten the input to an OS-independent (-1, 1)
292 z_delta = (z_delta < 0) ? -1 : 1;
293 input_process_mouse_wheel(z_delta);
294 }
295 } break;
296 case WM_LBUTTONDOWN:
297 case WM_MBUTTONDOWN:
298 case WM_RBUTTONDOWN:
299 case WM_LBUTTONUP:
300 case WM_MBUTTONUP:
301 case WM_RBUTTONUP: {
302 b8 pressed = msg == WM_LBUTTONDOWN || msg == WM_RBUTTONDOWN || msg == WM_MBUTTONDOWN;
303
304 // Not necessary but ynk
305 buttons mouse_button = BUTTON_MAX_BUTTONS;
306
307 switch (msg) {
308 case WM_LBUTTONDOWN:
309 case WM_LBUTTONUP:
310 mouse_button = BUTTON_LEFT;
311 break;
312 case WM_MBUTTONDOWN:
313 case WM_MBUTTONUP:
314 mouse_button = BUTTON_MIDDLE;
315 break;
316 case WM_RBUTTONDOWN:
317 case WM_RBUTTONUP:
318 mouse_button = BUTTON_RIGHT;
319 break;
320 }
321
322 // Pass over to the input subsystem.
323 if (mouse_button != BUTTON_MAX_BUTTONS) {
324 input_process_button(mouse_button, pressed);
325 }
326 } break;
327 }
328
329 return DefWindowProcA(hwnd, msg, w_param, l_param);
330}

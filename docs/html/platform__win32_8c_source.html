23// Initializes the platform: registers window class, creates a window, sets up the timer.
24b8 platform_startup(
25 platform_state *plat_state,
26 const char *application_name,
27 i32 x,
28 i32 y,
29 i32 width,
30 i32 height) {
31 // Allocate and set internal state
32 plat_state->internal_state = malloc(sizeof(internal_state));
33 internal_state *state = (internal_state *)plat_state->internal_state;
34 state->h_instance = GetModuleHandleA(0);
35
36 // Load default icon
37 HICON icon = LoadIcon(state->h_instance, IDI_APPLICATION);
38
39 // Setup and register window class.
40 WNDCLASSA wc;
41 memset(&wc, 0, sizeof(wc));
42 wc.style = CS_DBLCLKS; // Get double-clicks
43 wc.lpfnWndProc = win32_process_message;
44 wc.cbClsExtra = 0;
45 wc.cbWndExtra = 0;
46 wc.hInstance = state->h_instance;
47 wc.hIcon = icon;
48 wc.hCursor = LoadCursor(NULL, IDC_ARROW); // NULL; // Manage the cursor manually
49 wc.hbrBackground = NULL; // Transparent
50 wc.lpszClassName = "koru_window_class";
51
52 if (!RegisterClassA(&wc)) {
53 MessageBoxA(0, "Window registration failed", "Error", MB_ICONEXCLAMATION | MB_OK);
54 return FALSE;
55 }
56
57 // Desired client area dimensions
58 u32 client_x = x;
59 u32 client_y = y;
60 u32 client_width = width;
61 u32 client_height = height;
62
63 // These will be adjusted for window borders
64 u32 window_x = client_x;
65 u32 window_y = client_y;
66 u32 window_width = client_width;
67 u32 window_height = client_height;
68
69 // Window styles
70 u32 window_style = WS_OVERLAPPED | WS_SYSMENU | WS_CAPTION | WS_MAXIMIZEBOX | WS_MINIMIZEBOX | WS_THICKFRAME;
71 u32 window_ex_style = WS_EX_APPWINDOW;
72
73 // Adjust for window border to get correct client area size
74 RECT border_rect = {0, 0, 0, 0};
75 AdjustWindowRectEx(&border_rect, window_style, 0, window_ex_style);
76
77 // In this case, the border rectangle is negative.
78 window_x += border_rect.left;
79 window_y += border_rect.top;
80
81 // Grow by the size of the OS border.
82 window_width += border_rect.right - border_rect.left;
83 window_height += border_rect.bottom - border_rect.top;
84
85 HWND handle = CreateWindowExA(
86 window_ex_style, "koru_window_class", application_name,
87 window_style, window_x, window_y, window_width, window_height,
88 0, 0, state->h_instance, 0);
89
90 if (handle == 0) {
91 MessageBoxA(NULL, "Window creation failed!", "Error!", MB_ICONEXCLAMATION | MB_OK);
92
93 KFATAL("Window creation failed!");
94 return FALSE;
95 } else {
96 state->hwnd = handle;
97 }
98
99 // Show window (with input focus)
100 b32 should_activate = 1; // TODO: if the window should not accept input, this should be false.
101 i32 show_window_command_flags = should_activate ? SW_SHOW : SW_SHOWNOACTIVATE;
102 // If initially minimized, use SW_MINIMIZE : SW_SHOWMINNOACTIVE;
103 // If initially maximized, use SW_SHOWMAXIMIZED : SW_MAXIMIZE
104 ShowWindow(state->hwnd, show_window_command_flags);
105
106 // Setup high-performance timer
107 LARGE_INTEGER frequency;
108 QueryPerformanceFrequency(&frequency);
109 clock_frequency = 1.0 / (f64)frequency.QuadPart;
110 QueryPerformanceCounter(&start_time);
111
112 return TRUE;
113}

127// Processes all pending Windows messages (non-blocking)
128b8 platform_pump_messages(platform_state *plat_state) {
129 MSG message;
130
131 while (PeekMessageA(&message, NULL, 0, 0, PM_REMOVE)) {
132 TranslateMessage(&message);
133 DispatchMessageA(&message);
134 }
135
136 return TRUE;
137}

99b8 material_system_initialize(u64* memory_requirement, void* state, material_system_config config) {
100 if (config.max_material_count == 0) {
101 KFATAL("material_system_initialize - config.max_material_count must be > 0.");
102 return False;
103 }
104
105 // Block of memory will contain state structure, then block for array, then block for hashtable.
106 u64 struct_requirement = sizeof(material_system_state);
107 u64 array_requirement = sizeof(material) * config.max_material_count;
108 u64 hashtable_requirement = sizeof(material_reference) * config.max_material_count;
109 *memory_requirement = struct_requirement + array_requirement + hashtable_requirement;
110
111 if (!state) {
112 return True;
113 }
114
115 state_ptr = state;
116 state_ptr->config = config;
117
118 // The array block is after the state. Already allocated, so set the pointer.
119 void* array_block = state + struct_requirement;
120 state_ptr->registered_materials = array_block;
121
122 // Hashtable block is after array.
123 void* hashtable_block = array_block + array_requirement;
124
125 // Create a hashtable for material lookups.
126 hashtable_create(sizeof(material_reference), config.max_material_count, hashtable_block, False, &state_ptr->registered_material_table);
127
128 // Fill the hashtable with invalid references to use as a default.
129 material_reference invalid_ref;
130 invalid_ref.auto_release = False;
131 invalid_ref.handle = INVALID_ID; // Primary reason for needing default values.
132 invalid_ref.reference_count = 0;
133 hashtable_fill(&state_ptr->registered_material_table, &invalid_ref);
134
135 // Invalidate all materials in the array.
136 u32 count = state_ptr->config.max_material_count;
137 for (u32 i = 0; i < count; ++i) {
138 state_ptr->registered_materials[i].id = INVALID_ID;
139 state_ptr->registered_materials[i].generation = INVALID_ID;
140 state_ptr->registered_materials[i].internal_id = INVALID_ID;
141 }
142
143 if (!create_default_material(state_ptr)) {
144 KFATAL("Failed to create default material. Application cannot continue.");
145 return False;
146 }
147
148 return True;
149}

151void material_system_shutdown(void* state) {

153 if (s) {

158 destroy_material(&s->registered_materials[i]);

190material* material_system_acquire_from_config(material_config config) {
191 // Return default material.
192 if (strings_equali(config.name, DEFAULT_MATERIAL_NAME)) {
193 return &state_ptr->default_material;
194 }
195
196 material_reference ref;
197 if (state_ptr && hashtable_get(&state_ptr->registered_material_table, config.name, &ref)) {
198 // This can only be changed the first time a material is loaded.
199 if (ref.reference_count == 0) {
200 ref.auto_release = config.auto_release;
201 }
202
203 ref.reference_count++;
204
205 if (ref.handle == INVALID_ID) {
206 // This means no material exists here. Find a free index first.
207 u32 count = state_ptr->config.max_material_count;
208 material* m = 0;
209
210 for (u32 i = 0; i < count; ++i) {
211 if (state_ptr->registered_materials[i].id == INVALID_ID) {
212 // A free slot has been found. Use its index as the handle.
213 ref.handle = i;
214 m = &state_ptr->registered_materials[i];
215
216 break;
217 }
218 }
219
220 // Make sure an empty slot was actually found.
221 if (!m || ref.handle == INVALID_ID) {
222 KFATAL("material_system_acquire - Material system cannot hold anymore materials. Adjust configuration to allow more.");
223 return 0;
224 }
225
226 // Create new material.
227 if (!load_material(config, m)) {
228 KERROR("Failed to load material '%s'.", config.name);
229 return 0;
230 }
231
232 if (m->generation == INVALID_ID) {
233 m->generation = 0;
234 } else {
235 m->generation++;
236 }
237
238 // Also use the handle as the material id.
239 m->id = ref.handle;
240 KTRACE("Material '%s' does not yet exist. Created, and ref_count is now %i.", config.name, ref.reference_count);
241 } else {
242 KTRACE("Material '%s' already exists, ref_count increased to %i.", config.name, ref.reference_count);
243 }
244
245 // Update the entry.
246 hashtable_set(&state_ptr->registered_material_table, config.name, &ref);
247 return &state_ptr->registered_materials[ref.handle];
248 }
249
250 // NOTE: This would only happen in the event something went wrong with the state.
251 KERROR("material_system_acquire_from_config failed to acquire material '%s'. Null pointer will be returned.", config.name);
252 return 0;
253}

255void material_system_release(const char* name) {
256 // Ignore release requests for the default material.
257 if (strings_equali(name, DEFAULT_MATERIAL_NAME)) {
258 return;
259 }
260
261 material_reference ref;
262 if (state_ptr && hashtable_get(&state_ptr->registered_material_table, name, &ref)) {
263 if (ref.reference_count == 0) {
264 KWARN("Tried to release non-existent material: '%s'", name);
265 return;
266 }
267
268 ref.reference_count--;
269 if (ref.reference_count == 0 && ref.auto_release) {
270 material* m = &state_ptr->registered_materials[ref.handle];
271
272 // Destroy/reset material.
273 destroy_material(m);
274
275 // Reset the reference.
276 ref.handle = INVALID_ID;
277 ref.auto_release = False;
278 KTRACE("Released material '%s'., Material unloaded because reference count=0 and auto_release=true.", name);
279 } else {
280 KTRACE("Released material '%s', now has a reference count of '%i' (auto_release=%s).", name, ref.reference_count, ref.auto_release ? "true" : "false");
281 }
282
283 // Update the entry.
284 hashtable_set(&state_ptr->registered_material_table, name, &ref);
285 } else {
286 KERROR("material_system_release failed to release material '%s'.", name);
287 }
288}

364b8 load_configuration_file(const char* path, material_config* out_config) {
365 file_handle f;
366
367 // Only read from the file
368 if (!filesystem_open(path, FILE_MODE_READ, False, &f)) {
369 KERROR("load_configuration_file - unable to open material file for reading: '%s'.", path);
370 return False;
371 }
372
373 // Read each line of the file.
374 char line_buf[512] = "";
375 char* p = &line_buf[0];
376
377 u64 line_length = 0;
378 u32 line_number = 1;
379
380 while (filesystem_read_line(&f, 511, &p, &line_length)) {
381 // Trim the string.
382 char* trimmed = string_trim(line_buf);
383
384 // Get the trimmed length.
385 line_length = string_length(trimmed);
386
387 // Skip blank lines and comments.
388 if (line_length < 1 || trimmed[0] == '#') {
389 line_number++;
390 continue;
391 }
392
393 // Split into var/value
394 i32 equal_index = string_index_of(trimmed, '=');
395 if (equal_index == -1) {
396 KWARN("Potential formatting issue found in file '%s': '=' token not found. Skipping line %ui.", path, line_number);
397 line_number++;
398 continue;
399 }
400
401 // Assume a max of 64 characters for the variable name.
402 char raw_var_name[64];
403 kzero_memory(raw_var_name, sizeof(char) * 64);
404 string_mid(raw_var_name, trimmed, 0, equal_index);
405 char* trimmed_var_name = string_trim(raw_var_name);
406
407 // Assume a max of 511-65 (446) for the max length of the value to account for the variable name and the '='.
408 char raw_value[446];
409 kzero_memory(raw_value, sizeof(char) * 446);
410 string_mid(raw_value, trimmed, equal_index + 1, -1); // Read the rest of the line
411 char* trimmed_value = string_trim(raw_value);
412
413 // Process the variable.
414 if (strings_equali(trimmed_var_name, "version")) {
415 // TODO: version
416 } else if (strings_equali(trimmed_var_name, "name")) {
417 string_ncopy(out_config->name, trimmed_value, MATERIAL_NAME_MAX_LENGTH);
418 } else if (strings_equali(trimmed_var_name, "diffuse_map_name")) {
419 string_ncopy(out_config->diffuse_map_name, trimmed_value, TEXTURE_NAME_MAX_LENGTH);
420 } else if (strings_equali(trimmed_var_name, "diffuse_color")) {
421 // Parse the colour
422 if (!string_to_vec4(trimmed_value, &out_config->diffuse_color)) {
423 KWARN("Error parsing diffuse_color in file '%s'. Using default of white instead.", path);
424 out_config->diffuse_color = vec4_one(); // white
425 }
426 }
427
428 // TODO: more fields.
429
430 // Clear the line buffer.
431 kzero_memory(line_buf, sizeof(char) * 512);
432 line_number++;
433 }
434
435 filesystem_close(&f);
436
437 return True;
438}

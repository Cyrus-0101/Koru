504b8 physical_device_meets_requirements(
505 VkPhysicalDevice device,
506 VkSurfaceKHR surface,
507 const VkPhysicalDeviceProperties* properties,
508 const VkPhysicalDeviceFeatures* features,
509 const vulkan_physical_device_requirements* requirements,
510 vulkan_physical_device_queue_family_info* out_queue_info,
511 vulkan_swapchain_support_info* out_swapchain_support) {
512 // Evaluate device properties to determine if it meets the needs of our applcation.
513 out_queue_info->graphics_family_index = -1;
514 out_queue_info->present_family_index = -1;
515 out_queue_info->compute_family_index = -1;
516 out_queue_info->transfer_family_index = -1;
517
518 // Discrete GPU?
519 if (requirements->discrete_gpu) {
520 if (properties->deviceType != VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) {
521 KINFO("Device is not a discrete GPU, and one is required. Skipping.");
522 return FALSE;
523 }
524 }
525
526 u32 queue_family_count = 0;
527 vkGetPhysicalDeviceQueueFamilyProperties(device, &queue_family_count, 0);
528 VkQueueFamilyProperties queue_families[queue_family_count];
529 vkGetPhysicalDeviceQueueFamilyProperties(device, &queue_family_count, queue_families);
530
531 // Look at each queue and see what queues it supports
532 KINFO("Graphics | Present | Compute | Transfer | Name");
533 u8 min_transfer_score = 255;
534 for (u32 i = 0; i < queue_family_count; ++i) {
535 u8 current_transfer_score = 0;
536
537 // Graphics queue?
538 if (queue_families[i].queueFlags & VK_QUEUE_GRAPHICS_BIT) {
539 out_queue_info->graphics_family_index = i;
540 ++current_transfer_score;
541 }
542
543 // Compute queue?
544 if (queue_families[i].queueFlags & VK_QUEUE_COMPUTE_BIT) {
545 out_queue_info->compute_family_index = i;
546 ++current_transfer_score;
547 }
548
549 // Transfer queue?
550 if (queue_families[i].queueFlags & VK_QUEUE_TRANSFER_BIT) {
551 // Take the index if it is the current lowest. This increases the
552 // likelihood that it is a dedicated transfer queue.
553 if (current_transfer_score <= min_transfer_score) {
554 min_transfer_score = current_transfer_score;
555 out_queue_info->transfer_family_index = i;
556 }
557 }
558
559 // Present queue?
560 VkBool32 supports_present = VK_FALSE;
561 VK_CHECK(vkGetPhysicalDeviceSurfaceSupportKHR(device, i, surface, &supports_present));
562 if (supports_present) {
563 out_queue_info->present_family_index = i;
564 }
565 }
566
567 // Print out some info about the device
568 KINFO(" %d | %d | %d | %d | %s",
569 out_queue_info->graphics_family_index != -1,
570 out_queue_info->present_family_index != -1,
571 out_queue_info->compute_family_index != -1,
572 out_queue_info->transfer_family_index != -1,
573 properties->deviceName);
574
575 if (
576 (!requirements->graphics || (requirements->graphics && out_queue_info->graphics_family_index != -1)) &&
577 (!requirements->present || (requirements->present && out_queue_info->present_family_index != -1)) &&
578 (!requirements->compute || (requirements->compute && out_queue_info->compute_family_index != -1)) &&
579 (!requirements->transfer || (requirements->transfer && out_queue_info->transfer_family_index != -1))) {
580 KINFO("Device meets queue requirements.");
581 KTRACE("Graphics Family Index: %i", out_queue_info->graphics_family_index);
582 KTRACE("Present Family Index: %i", out_queue_info->present_family_index);
583 KTRACE("Transfer Family Index: %i", out_queue_info->transfer_family_index);
584 KTRACE("Compute Family Index: %i", out_queue_info->compute_family_index);
585
586 // Query swapchain support.
587 vulkan_device_query_swapchain_support(
588 device,
589 surface,
590 out_swapchain_support);
591
592 if (out_swapchain_support->format_count < 1 || out_swapchain_support->present_mode_count < 1) {
593 if (out_swapchain_support->formats) {
594 kfree(out_swapchain_support->formats, sizeof(VkSurfaceFormatKHR) * out_swapchain_support->format_count, MEMORY_TAG_RENDERER);
595 }
596 if (out_swapchain_support->present_modes) {
597 kfree(out_swapchain_support->present_modes, sizeof(VkPresentModeKHR) * out_swapchain_support->present_mode_count, MEMORY_TAG_RENDERER);
598 }
599 KINFO("Required swapchain support not present, skipping device.");
600 return FALSE;
601 }
602
603 // Device extensions.
604 if (requirements->device_extension_names) {
605 u32 available_extension_count = 0;
606 VkExtensionProperties* available_extensions = 0;
607 VK_CHECK(vkEnumerateDeviceExtensionProperties(
608 device,
609 0,
610 &available_extension_count,
611 0));
612 if (available_extension_count != 0) {
613 available_extensions = kallocate(sizeof(VkExtensionProperties) * available_extension_count, MEMORY_TAG_RENDERER);
614 VK_CHECK(vkEnumerateDeviceExtensionProperties(
615 device,
616 0,
617 &available_extension_count,
618 available_extensions));
619
620 u32 required_extension_count = darray_length(requirements->device_extension_names);
621 for (u32 i = 0; i < required_extension_count; ++i) {
622 b8 found = FALSE;
623 for (u32 j = 0; j < available_extension_count; ++j) {
624 if (strings_equal(requirements->device_extension_names[i], available_extensions[j].extensionName)) {
625 found = TRUE;
626 break;
627 }
628 }
629
630 if (!found) {
631 KINFO("Required extension not found: '%s', skipping device.", requirements->device_extension_names[i]);
632 kfree(available_extensions, sizeof(VkExtensionProperties) * available_extension_count, MEMORY_TAG_RENDERER);
633 return FALSE;
634 }
635 }
636 }
637 kfree(available_extensions, sizeof(VkExtensionProperties) * available_extension_count, MEMORY_TAG_RENDERER);
638 }
639
640 // Sampler anisotropy
641 if (requirements->sampler_anisotropy && !features->samplerAnisotropy) {
642 KINFO("Device does not support sampler_anisotropy, skipping.");
643 return FALSE;
644 }
645
646 // Device meets all requirements.
647 return TRUE;
648 }
649
650 return FALSE;

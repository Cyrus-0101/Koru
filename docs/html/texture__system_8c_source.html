98b8 texture_system_initialize(u64* memory_requirement, void* state, texture_system_config config) {
99 if (config.max_texture_count == 0) {
100 KFATAL("texture_system_initialize - config.max_texture_count must be > 0.");
101 return False;
102 }
103
104 // Block of memory will contain state structure, then block for array, then block for hashtable.
105 u64 struct_requirement = sizeof(texture_system_state);
106 u64 array_requirement = sizeof(texture) * config.max_texture_count;
107 u64 hashtable_requirement = sizeof(texture_reference) * config.max_texture_count;
108 *memory_requirement = struct_requirement + array_requirement + hashtable_requirement;
109
110 if (!state) {
111 return True;
112 }
113
114 state_ptr = state;
115 state_ptr->config = config;
116
117 // The array block is after the state. Already allocated, so just set the pointer.
118 void* array_block = state + struct_requirement;
119 state_ptr->registered_textures = array_block;
120
121 // Hashtable block is after array.
122 void* hashtable_block = array_block + array_requirement;
123
124 // Create a hashtable for texture lookups.
125 hashtable_create(sizeof(texture_reference), config.max_texture_count, hashtable_block, False, &state_ptr->registered_texture_table);
126
127 // Fill the hashtable with invalid references to use as a default.
128 texture_reference invalid_ref;
129 invalid_ref.auto_release = False;
130 invalid_ref.handle = INVALID_ID; // Primary reason for needing default values.
131 invalid_ref.reference_count = 0;
132 hashtable_fill(&state_ptr->registered_texture_table, &invalid_ref);
133
134 // Invalidate all textures in the array.
135 u32 count = state_ptr->config.max_texture_count;
136 for (u32 i = 0; i < count; ++i) {
137 state_ptr->registered_textures[i].id = INVALID_ID;
138 state_ptr->registered_textures[i].generation = INVALID_ID;
139 }
140
141 // Create default textures for use in the system.
142 create_default_textures(state_ptr);
143
144 return True;
145}

163texture* texture_system_acquire(const char* name, b8 auto_release) {
164 // Return default texture, but warn about it since this should be returned via get_default_texture();
165 if (strings_equali(name, DEFAULT_TEXTURE_NAME)) {
166 KWARN("texture_system_acquire called for default texture. Use texture_system_get_default_texture for texture 'default'.");
167 return &state_ptr->default_texture;
168 }
169
170 texture_reference ref;
171 if (state_ptr && hashtable_get(&state_ptr->registered_texture_table, name, &ref)) {
172 // This can only be changed the first time a texture is loaded.
173 if (ref.reference_count == 0) {
174 ref.auto_release = auto_release;
175 }
176 ref.reference_count++;
177 if (ref.handle == INVALID_ID) {
178 // This means no texture exists here. Find a free index first.
179 u32 count = state_ptr->config.max_texture_count;
180 texture* t = 0;
181 for (u32 i = 0; i < count; ++i) {
182 if (state_ptr->registered_textures[i].id == INVALID_ID) {
183 // A free slot has been found. Use its index as the handle.
184 ref.handle = i;
185 t = &state_ptr->registered_textures[i];
186 break;
187 }
188 }
189
190 // Make sure an empty slot was actually found.
191 if (!t || ref.handle == INVALID_ID) {
192 KFATAL("texture_system_acquire - Texture system cannot hold anymore textures. Adjust configuration to allow more.");
193 return 0;
194 }
195
196 // Create new texture.
197 if (!load_texture(name, t)) {
198 KERROR("Failed to load texture '%s'.", name);
199 return 0;
200 }
201
202 // Also use the handle as the texture id.
203 t->id = ref.handle;
204 KTRACE("Texture '%s' does not yet exist. Created, and ref_count is now %i.", name, ref.reference_count);
205 } else {
206 KTRACE("Texture '%s' already exists, ref_count increased to %i.", name, ref.reference_count);
207 }
208
209 // Update the entry.
210 hashtable_set(&state_ptr->registered_texture_table, name, &ref);
211 return &state_ptr->registered_textures[ref.handle];
212 }
213
214 // NOTE: This would only happen in the event something went wrong with the state.
215 KERROR("texture_system_acquire failed to acquire texture '%s'. Null pointer will be returned.", name);
216 return 0;
217}

219void texture_system_release(const char* name) {
220 // Ignore release requests for the default texture.
221 if (strings_equali(name, DEFAULT_TEXTURE_NAME)) {
222 return;
223 }
224 texture_reference ref;
225 if (state_ptr && hashtable_get(&state_ptr->registered_texture_table, name, &ref)) {
226 if (ref.reference_count == 0) {
227 KWARN("Tried to release non-existent texture: '%s'", name);
228 return;
229 }
230 ref.reference_count--;
231 if (ref.reference_count == 0 && ref.auto_release) {
232 texture* t = &state_ptr->registered_textures[ref.handle];
233
234 // Release texture.
235 renderer_destroy_texture(t);
236
237 // Reset the array entry, ensure invalid ids are set.
238 kzero_memory(t, sizeof(texture));
239 t->id = INVALID_ID;
240 t->generation = INVALID_ID;
241
242 // Reset the reference.
243 ref.handle = INVALID_ID;
244 ref.auto_release = False;
245 KTRACE("Released texture '%s'., Texture unloaded because reference count=0 and auto_release=true.", name);
246 } else {
247 KTRACE("Released texture '%s', now has a reference count of '%i' (auto_release=%s).", name, ref.reference_count, ref.auto_release ? "true" : "false");
248 }
249
250 // Update the entry.
251 hashtable_set(&state_ptr->registered_texture_table, name, &ref);
252 } else {
253 KERROR("texture_system_release failed to release texture '%s'.", name);
254 }

266b8 create_default_textures(texture_system_state* state) {
267 // NOTE: Create default texture, a 256x256 blue/white checkerboard pattern.
268 // This is done in code to eliminate asset dependencies.
269 KTRACE("Creating default texture...");
270
271 u8 pixels[PIXEL_COUNT * CHANNELS];
272
273 kset_memory(pixels, 255, sizeof(u8) * PIXEL_COUNT * CHANNELS);
274
275 // Each pixel.
276 for (u64 row = 0; row < TEX_DIMENSIONS; ++row) {
277 for (u64 col = 0; col < TEX_DIMENSIONS; ++col) {
278 u64 index = (row * TEX_DIMENSIONS) + col;
279 u64 index_bpp = index * CHANNELS;
280
281 if (row % 2) {
282 if (col % 2) {
283 pixels[index_bpp + 0] = 0;
284 pixels[index_bpp + 1] = 0;
285 }
286 } else {
287 if (!(col % 2)) {
288 pixels[index_bpp + 0] = 0;
289 pixels[index_bpp + 1] = 0;
290 }
291 }
292 }
293 }
294
295 renderer_create_texture(DEFAULT_TEXTURE_NAME, TEX_DIMENSIONS, TEX_DIMENSIONS, 4, pixels, False, &state->default_texture);
296 // Manually set the texture generation to invalid since this is a default texture.
297 state->default_texture.generation = INVALID_ID;
298
299 return True;
300}

308b8 load_texture(const char* texture_name, texture* t) {
309 // TODO: Should be able to be located anywhere.
310 char* format_str = "assets/textures/%s.%s";
311 const i32 required_channel_count = 4;
312 stbi_set_flip_vertically_on_load(True);
313 char full_file_path[512];
314
315 // TODO: try different extensions
316 string_format(full_file_path, format_str, texture_name, "png");
317
318 // Use a temporary texture to load into.
319 texture temp_texture;
320
321 u8* data = stbi_load(
322 full_file_path,
323 (i32*)&temp_texture.width,
324 (i32*)&temp_texture.height,
325 (i32*)&temp_texture.channel_count,
326 required_channel_count);
327
328 temp_texture.channel_count = required_channel_count;
329
330 if (data) {
331 u32 current_generation = t->generation;
332 t->generation = INVALID_ID;
333
334 u64 total_size = temp_texture.width * temp_texture.height * required_channel_count;
335 // Check for transparency
336 b32 has_transparency = False;
337 for (u64 i = 0; i < total_size; i += required_channel_count) {
338 u8 a = data[i + 3];
339 if (a < 255) {
340 has_transparency = True;
341 break;
342 }
343 }
344
345 if (stbi_failure_reason()) {
346 KWARN("load_texture() failed to load file '%s': %s", full_file_path, stbi_failure_reason());
347 }
348
349 // Acquire internal texture resources and upload to GPU.
350 renderer_create_texture(
351 texture_name,
352 temp_texture.width,
353 temp_texture.height,
354 temp_texture.channel_count,
355 data,
356 has_transparency,
357 &temp_texture);
358
359 // Take a copy of the old texture.
360 texture old = *t;
361
362 // Assign the temp texture to the pointer.
363 *t = temp_texture;
364
365 // Destroy the old texture.
366 renderer_destroy_texture(&old);
367
368 if (current_generation == INVALID_ID) {
369 t->generation = 0;
370 } else {
371 t->generation = current_generation + 1;
372 }
373
374 // Clean up data.
375 stbi_image_free(data);
376 return True;
377 } else {
378 if (stbi_failure_reason()) {
379 KWARN("load_texture() failed to load file '%s': %s", full_file_path, stbi_failure_reason());
380 }
381 return False;
382 }

122b8 texture_system_initialize(u64* memory_requirement, void* state, texture_system_config config) {
123 if (config.max_texture_count == 0) {
124 KFATAL("texture_system_initialize - config.max_texture_count must be > 0.");
125 return False;
126 }
127
128 // Block of memory will contain state structure, then block for array, then block for hashtable.
129 u64 struct_requirement = sizeof(texture_system_state);
130 u64 array_requirement = sizeof(texture) * config.max_texture_count;
131 u64 hashtable_requirement = sizeof(texture_reference) * config.max_texture_count;
132 *memory_requirement = struct_requirement + array_requirement + hashtable_requirement;
133
134 if (!state) {
135 return True;
136 }
137
138 state_ptr = state;
139 state_ptr->config = config;
140
141 // The array block is after the state. Already allocated, so just set the pointer.
142 void* array_block = state + struct_requirement;
143 state_ptr->registered_textures = array_block;
144
145 // Hashtable block is after array.
146 void* hashtable_block = array_block + array_requirement;
147
148 // Create a hashtable for texture lookups.
149 hashtable_create(sizeof(texture_reference), config.max_texture_count, hashtable_block, False, &state_ptr->registered_texture_table);
150
151 // Fill the hashtable with invalid references to use as a default.
152 texture_reference invalid_ref;
153 invalid_ref.auto_release = False;
154 invalid_ref.handle = INVALID_ID; // Primary reason for needing default values.
155 invalid_ref.reference_count = 0;
156 hashtable_fill(&state_ptr->registered_texture_table, &invalid_ref);
157
158 // Invalidate all textures in the array.
159 u32 count = state_ptr->config.max_texture_count;
160 for (u32 i = 0; i < count; ++i) {
161 state_ptr->registered_textures[i].id = INVALID_ID;
162 state_ptr->registered_textures[i].generation = INVALID_ID;
163 }
164
165 // Create default textures for use in the system.
166 create_default_textures(state_ptr);
167
168 return True;
169}

172 if (state_ptr) {
173 // Destroy all loaded textures.
174 for (u32 i = 0; i < state_ptr->config.max_texture_count; ++i) {

179 }

187texture* texture_system_acquire(const char* name, b8 auto_release) {
188 // Return default texture, but warn about it since this should be returned via get_default_texture();
189 if (strings_equali(name, DEFAULT_TEXTURE_NAME)) {
190 KWARN("texture_system_acquire called for default texture. Use texture_system_get_default_texture for texture 'default'.");
191 return &state_ptr->default_texture;
192 }
193
194 texture_reference ref;
195 if (state_ptr && hashtable_get(&state_ptr->registered_texture_table, name, &ref)) {
196 // This can only be changed the first time a texture is loaded.
197 if (ref.reference_count == 0) {
198 ref.auto_release = auto_release;
199 }
200 ref.reference_count++;
201 if (ref.handle == INVALID_ID) {
202 // This means no texture exists here. Find a free index first.
203 u32 count = state_ptr->config.max_texture_count;
204 texture* t = 0;
205 for (u32 i = 0; i < count; ++i) {
206 if (state_ptr->registered_textures[i].id == INVALID_ID) {
207 // A free slot has been found. Use its index as the handle.
208 ref.handle = i;
209 t = &state_ptr->registered_textures[i];
210 break;
211 }
212 }
213
214 // Make sure an empty slot was actually found.
215 if (!t || ref.handle == INVALID_ID) {
216 KFATAL("texture_system_acquire - Texture system cannot hold anymore textures. Adjust configuration to allow more.");
217 return 0;
218 }
219
220 // Create new texture.
221 if (!load_texture(name, t)) {
222 KERROR("Failed to load texture '%s'.", name);
223 return 0;
224 }
225
226 // Also use the handle as the texture id.
227 t->id = ref.handle;
228 KTRACE("Texture '%s' does not yet exist. Created, and ref_count is now %i.", name, ref.reference_count);
229 } else {
230 KTRACE("Texture '%s' already exists, ref_count increased to %i.", name, ref.reference_count);
231 }
232
233 // Update the entry.
234 hashtable_set(&state_ptr->registered_texture_table, name, &ref);
235 return &state_ptr->registered_textures[ref.handle];
236 }
237
238 // NOTE: This would only happen in the event something went wrong with the state.
239 KERROR("texture_system_acquire failed to acquire texture '%s'. Null pointer will be returned.", name);
240 return 0;
241}

243void texture_system_release(const char* name) {
244 // Ignore release requests for the default texture.
245 if (strings_equali(name, DEFAULT_TEXTURE_NAME)) {
246 return;
247 }
248 texture_reference ref;
249 if (state_ptr && hashtable_get(&state_ptr->registered_texture_table, name, &ref)) {
250 if (ref.reference_count == 0) {
251 KWARN("Tried to release non-existent texture: '%s'", name);
252 return;
253 }
254
255 // Take a copy of the name since it will be wiped out by destroy,
256 // (as passed in name is generally a pointer to the actual texture's name).
257 char name_copy[TEXTURE_NAME_MAX_LENGTH];
258 string_ncopy(name_copy, name, TEXTURE_NAME_MAX_LENGTH);
259
260 ref.reference_count--;
261
262 if (ref.reference_count == 0 && ref.auto_release) {
263 texture* t = &state_ptr->registered_textures[ref.handle];
264
265 // Destroy/Reset texture.
266 destroy_texture(t);
267
268 // Reset the reference.
269 ref.handle = INVALID_ID;
270 ref.auto_release = False;
271 KTRACE("Released texture '%s'., Texture unloaded because reference count=0 and auto_release=true.", name);
272 } else {
273 KTRACE("Released texture '%s', now has a reference count of '%i' (auto_release=%s).", name_copy, ref.reference_count, ref.auto_release ? "true" : "false");
274 }
275
276 // Update the entry.
277 hashtable_set(&state_ptr->registered_texture_table, name_copy, &ref);
278 } else {
279 KERROR("texture_system_release failed to release texture '%s'.", name);
280 }
281}

292b8 create_default_textures(texture_system_state* state) {
293 // NOTE: Create default texture, a 256x256 blue/white checkerboard pattern.
294 // This is done in code to eliminate asset dependencies.
295 KTRACE("Creating default texture...");
296
297 u8 pixels[PIXEL_COUNT * CHANNELS];
298
299 kset_memory(pixels, 255, sizeof(u8) * PIXEL_COUNT * CHANNELS);
300
301 // Each pixel.
302 for (u64 row = 0; row < TEX_DIMENSIONS; ++row) {
303 for (u64 col = 0; col < TEX_DIMENSIONS; ++col) {
304 u64 index = (row * TEX_DIMENSIONS) + col;
305 u64 index_bpp = index * CHANNELS;
306
307 if (row % 2) {
308 if (col % 2) {
309 pixels[index_bpp + 0] = 0;
310 pixels[index_bpp + 1] = 0;
311 }
312 } else {
313 if (!(col % 2)) {
314 pixels[index_bpp + 0] = 0;
315 pixels[index_bpp + 1] = 0;
316 }
317 }
318 }
319 }
320
321 string_ncopy(state->default_texture.name, DEFAULT_TEXTURE_NAME, TEXTURE_NAME_MAX_LENGTH);
322 state->default_texture.width = TEX_DIMENSIONS;
323 state->default_texture.height = TEX_DIMENSIONS;
324 state->default_texture.channel_count = 4;
325 state->default_texture.generation = INVALID_ID;
326 state->default_texture.has_transparency = False;
327
328 renderer_create_texture(DEFAULT_TEXTURE_NAME, TEX_DIMENSIONS, TEX_DIMENSIONS, 4, pixels, False, &state->default_texture);
329 // Manually set the texture generation to invalid since this is a default texture.
330 state->default_texture.generation = INVALID_ID;
331
332 return True;
333}

341b8 load_texture(const char* texture_name, texture* t) {
342 // TODO: Should be able to be located anywhere.
343 char* format_str = "assets/textures/%s.%s";
344 const i32 required_channel_count = 4;
345 stbi_set_flip_vertically_on_load(True);
346 char full_file_path[512];
347
348 // TODO: try different extensions
349 string_format(full_file_path, format_str, texture_name, "png");
350
351 // Use a temporary texture to load into.
352 texture temp_texture;
353
354 u8* data = stbi_load(
355 full_file_path,
356 (i32*)&temp_texture.width,
357 (i32*)&temp_texture.height,
358 (i32*)&temp_texture.channel_count,
359 required_channel_count);
360
361 temp_texture.channel_count = required_channel_count;
362
363 if (data) {
364 u32 current_generation = t->generation;
365 t->generation = INVALID_ID;
366
367 u64 total_size = temp_texture.width * temp_texture.height * required_channel_count;
368 // Check for transparency
369 b32 has_transparency = False;
370 for (u64 i = 0; i < total_size; i += required_channel_count) {
371 u8 a = data[i + 3];
372 if (a < 255) {
373 has_transparency = True;
374 break;
375 }
376 }
377
378 if (stbi_failure_reason()) {
379 KWARN("load_texture() failed to load file '%s': %s", full_file_path, stbi_failure_reason());
380 }
381
382 // Acquire internal texture resources and upload to GPU.
383 renderer_create_texture(
384 texture_name,
385 temp_texture.width,
386 temp_texture.height,
387 temp_texture.channel_count,
388 data,
389 has_transparency,
390 &temp_texture);
391
392 // Take a copy of the old texture.
393 texture old = *t;
394
395 // Assign the temp texture to the pointer.
396 *t = temp_texture;
397
398 // Destroy the old texture.
399 renderer_destroy_texture(&old);
400
401 if (current_generation == INVALID_ID) {
402 t->generation = 0;
403 } else {
404 t->generation = current_generation + 1;
405 }
406
407 // Clean up data.
408 stbi_image_free(data);
409 return True;
410 } else {
411 if (stbi_failure_reason()) {
412 KWARN("load_texture() failed to load file '%s': %s", full_file_path, stbi_failure_reason());
413 }
414 return False;
415 }

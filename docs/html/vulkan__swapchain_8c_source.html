158void vulkan_swapchain_present(
159 vulkan_context* context,
160 vulkan_swapchain* swapchain,
161 VkQueue graphics_queue,
162 VkQueue present_queue,
163 VkSemaphore render_complete_semaphore,
164 u32 present_image_index) {
165 // Return the image to the swapchain for presentation.
166 VkPresentInfoKHR present_info = {VK_STRUCTURE_TYPE_PRESENT_INFO_KHR};
167 present_info.waitSemaphoreCount = 1;
168 present_info.pWaitSemaphores = &render_complete_semaphore;
169 present_info.swapchainCount = 1;
170 present_info.pSwapchains = &swapchain->handle;
171 present_info.pImageIndices = &present_image_index;
172 present_info.pResults = 0;
173
174 VkResult result = vkQueuePresentKHR(present_queue, &present_info);
175 if (result == VK_ERROR_OUT_OF_DATE_KHR || result == VK_SUBOPTIMAL_KHR) {
176 // Swapchain is out of date, suboptimal or a framebuffer resize has occurred. Trigger swapchain recreation.
177 vulkan_swapchain_recreate(context, context->framebuffer_width, context->framebuffer_height, swapchain);
178 } else if (result != VK_SUCCESS) {
179 KFATAL("Failed to present swapchain image!");
180 }
181
182 // Increment (and loop) the index.
183 context->current_frame = (context->current_frame + 1) % swapchain->max_frames_in_flight;
184}

205 for (u32 i = 0; i < context->device.swapchain_support.format_count; ++i) {
206 VkSurfaceFormatKHR format = context->device.swapchain_support.formats[i];
207 // Preferred formats: B8G8R8A8_UNORM + sRGB color space
208 if (format.format == VK_FORMAT_B8G8R8A8_UNORM &&
209 format.colorSpace == VK_COLOR_SPACE_SRGB_NONLINEAR_KHR) {
210 swapchain->image_format = format;
211 found = True;
212 break;
213 }
214 }
215
216 if (!found) {
217 swapchain->image_format = context->device.swapchain_support.formats[0];
218 }
219
220 // Choose a present mode
221 VkPresentModeKHR present_mode = VK_PRESENT_MODE_FIFO_KHR; // Default: vsync
222 for (u32 i = 0; i < context->device.swapchain_support.present_mode_count; ++i) {
223 VkPresentModeKHR mode = context->device.swapchain_support.present_modes[i];
224 if (mode == VK_PRESENT_MODE_MAILBOX_KHR) {
225 present_mode = mode;
226 break;
227 }
228 }
229
230 // Requery swapchain support
231 vulkan_device_query_swapchain_support(
232 context->device.physical_device,
233 context->surface,
234 &context->device.swapchain_support);
235
236 // Set the correct extent
237 if (context->device.swapchain_support.capabilities.currentExtent.width != UINT32_MAX) {
238 swapchain_extent = context->device.swapchain_support.capabilities.currentExtent;
239 }
240
241 // Clamp to min/max allowed by the device
242 VkExtent2D min = context->device.swapchain_support.capabilities.minImageExtent;
243 VkExtent2D max = context->device.swapchain_support.capabilities.maxImageExtent;
244 swapchain_extent.width = KCLAMP(swapchain_extent.width, min.width, max.width);
245 swapchain_extent.height = KCLAMP(swapchain_extent.height, min.height, max.height);
246
247 // Determine image count (one more than minimum, but not exceeding maximum)
248 u32 image_count = context->device.swapchain_support.capabilities.minImageCount + 1;
249 if (context->device.swapchain_support.capabilities.maxImageCount > 0 && image_count > context->device.swapchain_support.capabilities.maxImageCount) {

251 }

257 swapchain_create_info.surface = context->surface;
258 swapchain_create_info.minImageCount = image_count;
259 swapchain_create_info.imageFormat = swapchain->image_format.format;
260 swapchain_create_info.imageColorSpace = swapchain->image_format.colorSpace;
261 swapchain_create_info.imageExtent = swapchain_extent;
262 swapchain_create_info.imageArrayLayers = 1;
263 swapchain_create_info.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
264
265 // Setup image sharing mode & the queue family indices
266 if (context->device.graphics_queue_index != context->device.present_queue_index) {
267 u32 queueFamilyIndices[] = {
268 (u32)context->device.graphics_queue_index,
269 (u32)context->device.present_queue_index};
270 swapchain_create_info.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
271 swapchain_create_info.queueFamilyIndexCount = 2;
272 swapchain_create_info.pQueueFamilyIndices = queueFamilyIndices;
273 } else {
274 swapchain_create_info.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
275 swapchain_create_info.queueFamilyIndexCount = 0;
276 swapchain_create_info.pQueueFamilyIndices = 0;
277 }

281 swapchain_create_info.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
282 swapchain_create_info.presentMode = present_mode;
283 swapchain_create_info.clipped = VK_TRUE;
284 swapchain_create_info.oldSwapchain = 0;

288
289 // Reset frame index
290 context->current_frame = 0;
291
292 // Get images
293 swapchain->image_count = 0;

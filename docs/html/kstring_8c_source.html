84char* string_trim(char* str) {
85 // Checks if each character is a whitespace character and
86 // moves the pointer forward until a non-whitespace character is found.
87 while (isspace((unsigned char)*str)) {
88 str++;
89 }
90 // If the string is not empty after trimming leading whitespace,
91 // we find the end of the string and move backwards to trim trailing whitespace.
92 if (*str) {
93 char* p = str;
94 // Move to the end of the string.
95 while (*p) {
96 p++;
97 }
98
99 // Move backwards over any trailing whitespace.
100 while (isspace((unsigned char)*(--p))); // Dereference check if space after decrement.
101
102 p[1] = '\0';
103 }
104
105 return str;
106}

108void string_mid(char* dest, const char* source, i32 start, i32 length) {
109 if (length == 0) {
110 return;
111 }
112
113 u64 src_length = string_length(source);
114 if (start >= src_length) {
115 dest[0] = 0;
116 return;
117 }
118 if (length > 0) {
119 for (u64 i = start, j = 0; j < length && source[i]; ++i, ++j) {
120 dest[j] = source[i];

122 dest[start + length] = 0;
123 } else {
124 // If a negative value is passed, proceed to the end of the string.
125 u64 j = 0;
126 for (u64 i = start; source[i]; ++i, ++j) {
127 dest[j] = source[i];
128 }
129 dest[start + j] = 0;
130 }

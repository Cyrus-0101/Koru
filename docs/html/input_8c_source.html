139void input_process_key(keys key, b8 pressed) {
140 // Only handle changes in state
141 if (state_ptr && state_ptr->keyboard_current.keys[key] != pressed) {
142 // Update internal state_ptr->
143 state_ptr->keyboard_current.keys[key] = pressed;

145 if (key == KEY_LALT) {

154 KINFO("Right Shift Key Pressed");
155 }
156
157 if (key == KEY_LCONTROL) {
158 KINFO("Left Control Key Pressed");
159 } else if (key == KEY_RCONTROL) {
160 KINFO("Right Control Key Pressed");
161 }
162
163 // Update internal state
164 state_ptr->keyboard_current.keys[key] = pressed;
165

167 event_context context;
168 context.data.u16[0] = key;

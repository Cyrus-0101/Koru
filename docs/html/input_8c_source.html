130void input_process_key(keys key, b8 pressed) {
131 // Only handle changes in state
132 if (state.keyboard_current.keys[key] != pressed) {
133 if (key == KEY_LALT) {
134 KINFO("Left Alt Key Pressed");
135 } else if (key == KEY_RALT) {
136 KINFO("Right Alt Key Pressed");
137 } else if (key == KEY_LSHIFT) {
138 KINFO("Left Shift Key Pressed");
139 } else if (key == KEY_RSHIFT) {
140 KINFO("Right Shift Key Pressed");
141 } else if (key == KEY_LCONTROL) {
142 KINFO("Left Control Key Pressed");
143 } else if (key == KEY_RCONTROL) {
144 KINFO("Right Control Key Pressed");
145 }
146
147 // Update internal state
148 state.keyboard_current.keys[key] = pressed;
149
150 // Fire event so listeners can respond
151 event_context context;
152 context.data.u16[0] = key;
153 event_fire(pressed ? EVENT_CODE_KEY_PRESSED : EVENT_CODE_KEY_RELEASED, 0, context);
154 }
155}

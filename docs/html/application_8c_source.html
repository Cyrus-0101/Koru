425b8 application_run() {
426 app_state->is_running = True;
427
428 clock_start(&app_state->clock);
429
430 clock_update(&app_state->clock);
431
432 app_state->last_time = app_state->clock.elapsed;
433
434 f64 running_time = 0;
435
436 u8 frame_count = 0;
437
438 f64 target_frame_seconds = 1.0f / 60;
439
440 // Log memory info - Memory Leak
441 KINFO(get_memory_usage_str());
442
443 while (app_state->is_running) {
444 if (!platform_pump_messages()) {
445 // If platform request to quit, stop running
446 app_state->is_running = False;
447 }
448
449 if (!app_state->is_suspended) {
450 // Update clock and get delta time.
451 clock_update(&app_state->clock);
452
453 f64 current_time = app_state->clock.elapsed;
454
455 f64 delta = (current_time - app_state->last_time);
456
457 f64 frame_start_time = platform_get_absolute_time();
458
459 if (!app_state->game_inst->update(app_state->game_inst, (f32)delta)) {
460 KFATAL("Game update failed. Shutting down!");
461
462 app_state->is_running = False;
463
464 break;
465 }
466
467 // Call game render routine
468 if (!app_state->game_inst->render(app_state->game_inst, (f32)delta)) {
469 KFATAL("Game render failed. Shutting down!");
470
471 app_state->is_running = False;
472
473 break;
474 }
475
476 // TODO: refactor packet creation
477 render_packet packet;
478
479 packet.delta_time = delta;
480
481 // TODO: temp
482 geometry_render_data test_render;
483 test_render.geometry = app_state->test_geometry;
484 test_render.model = mat4_identity();
485
486 packet.geometry_count = 1;
487 packet.geometries = &test_render;
488 // TODO: end temp
489
490 renderer_draw_frame(&packet);
491
492 // Figure out how long the frame took and, if below
493 f64 frame_end_time = platform_get_absolute_time();
494
495 f64 frame_elapsed_time = frame_end_time - frame_start_time;
496
497 running_time += frame_elapsed_time;
498
499 f64 remaining_seconds = target_frame_seconds - frame_elapsed_time;
500
501 if (remaining_seconds > 0) {
502 u64 remaining_ms = (remaining_seconds * 1000);
503
504 // If there is time left, give it back to the OS.
505 b8 limit_frames = False;
506
507 if (remaining_ms > 0 && limit_frames) {
508 platform_sleep(remaining_ms - 1);
509 }
510
511 frame_count++;
512 }
513
514 // NOTE: Input update/state copying should always be handled
515 // after any input should be recorded; I.E. before this line.
516 // As a safety, input is the last thing to be updated before
517 // this frame ends.
518 input_update(delta);
519
520 // Update last time
521 app_state->last_time = current_time;
522 }
523 }
524
525 // Ensure running state is cleared
526 app_state->is_running = False;
527
528 // Shutdown event system.
529 event_unregister(EVENT_CODE_APPLICATION_QUIT, 0, application_on_event);
530 event_unregister(EVENT_CODE_KEY_PRESSED, 0, application_on_key);
531 event_unregister(EVENT_CODE_KEY_RELEASED, 0, application_on_key);

533 // TODO: Temp code
534 event_unregister(EVENT_CODE_DEBUG0, 0, event_on_debug_event);
535 // TODO: End temp code

538

574b8 application_on_key(u16 code, void* sender, void* listener_inst, event_context context) {
575 if (code == EVENT_CODE_KEY_PRESSED) {
576 u16 key_code = context.data.u16[0];
577 if (key_code == KEY_ESCAPE) {
578 // NOTE: Technically firing an event to itself, but there may be other listeners.
579 event_context data = {};
580 event_fire(EVENT_CODE_APPLICATION_QUIT, 0, data);
581
582 // Block anything else from processing this.
583 return True;
584 } else if (key_code == KEY_A) {
585 // Example on checking for a key
586 KDEBUG("Explicit - A key pressed!");
587 } else {
588 KDEBUG("'%c' key pressed in window.", key_code);
589 }
590 } else if (code == EVENT_CODE_KEY_RELEASED) {
591 u16 key_code = context.data.u16[0];
592 if (key_code == KEY_B) {
593 // Example on checking for a key
594 KDEBUG("Explicit - B key released!");
595 } else {
596 KDEBUG("'%c' key released in window.", key_code);
597 }
598 }
599 return False;
600}

602b8 application_on_resized(u16 code, void* sender, void* listener_inst, event_context context) {
603 if (code == EVENT_CODE_RESIZED) {
604 u16 width = context.data.u16[0];
605 u16 height = context.data.u16[1];
606
607 if (width != app_state->width || height != app_state->height) {
608 app_state->width = width;
609 app_state->height = height;
610
611 KDEBUG("Window resize: %i, %i", width, height);
612
613 // Handle minimization
614 if (width == 0 || height == 0) {
615 KINFO("Window minimized, suspending application.");
616 app_state->is_suspended = True;
617 return True;
618 } else {
619 if (app_state->is_suspended) {
620 KINFO("Window restored, resuming application.");
621 app_state->is_suspended = False;
622 }
623 app_state->game_inst->on_resize(app_state->game_inst, width, height);
624 renderer_on_resized(width, height);
625 }
626 }
627 }
628 return False;

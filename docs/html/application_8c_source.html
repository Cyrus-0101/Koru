170b8 application_run() {
171 // Log memory info - Memory Leak
172 KINFO(get_memory_usage_str());
173
174 while (app_state.is_running) {
175 if (!platform_pump_messages(&app_state.platform)) {
176 // If platform request to quit, stop running
177 app_state.is_running = FALSE;
178 break;
179 }
180
181 if (!app_state.is_suspended) {
182 if (!app_state.game_inst->update(app_state.game_inst, (f32)0)) {
183 KFATAL("Game update failed. Shutting down!");
184 app_state.is_running = FALSE;
185
186 break;
187 }
188
189 // Call game render routine
190 if (!app_state.game_inst->render(app_state.game_inst, (f32)0)) {
191 KFATAL("Game render failed. Shutting down!");
192 app_state.is_running = FALSE;
193
194 break;
195 }
196
197 // NOTE: Input update/state copying should always be handled
198 // after any input should be recorded; I.E. before this line.
199 // As a safety, input is the last thing to be updated before
200 // this frame ends.
201 input_update(0);
202 }
203 }
204
205 // Ensure running state is cleared
206 app_state.is_running = FALSE;
207
208 // Shutdown event system.
209 event_unregister(EVENT_CODE_APPLICATION_QUIT, 0, application_on_event);
210 event_unregister(EVENT_CODE_KEY_PRESSED, 0, application_on_key);
211 event_unregister(EVENT_CODE_KEY_RELEASED, 0, application_on_key);
212
213 event_shutdown();
214 input_shutdown();
215
216 // Clean up platform resources
217 platform_shutdown(&app_state.platform);

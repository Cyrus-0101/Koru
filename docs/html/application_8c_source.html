173b8 application_run() {
174 clock_start(&app_state.clock);
175
176 clock_update(&app_state.clock);
177
178 app_state.last_time = app_state.clock.elapsed;
179
180 f64 running_time = 0;
181
182 u8 frame_count = 0;
183
184 f64 target_frame_seconds = 1.0f / 60;
185
186 // Log memory info - Memory Leak
187 KINFO(get_memory_usage_str());
188
189 while (app_state.is_running) {
190 if (!platform_pump_messages(&app_state.platform)) {
191 // If platform request to quit, stop running
192 app_state.is_running = FALSE;
193 break;
194 }
195
196 if (!app_state.is_suspended) {
197 // Update clock and get delta time.
198 clock_update(&app_state.clock);
199
200 f64 current_time = app_state.clock.elapsed;
201
202 f64 delta = (current_time - app_state.last_time);
203
204 f64 frame_start_time = platform_get_absolute_time();
205
206 if (!app_state.game_inst->update(app_state.game_inst, (f32)delta)) {
207 KFATAL("Game update failed. Shutting down!");
208
209 app_state.is_running = FALSE;
210
211 break;
212 }
213
214 // Call game render routine
215 if (!app_state.game_inst->render(app_state.game_inst, (f32)delta)) {
216 KFATAL("Game render failed. Shutting down!");
217 app_state.is_running = FALSE;
218
219 break;
220 }
221
222 // TODO: refactor packet creation
223 render_packet packet;
224
225 packet.delta_time = delta;
226
227 renderer_draw_frame(&packet);
228
229 // Figure out how long the frame took and, if below
230 f64 frame_end_time = platform_get_absolute_time();
231
232 f64 frame_elapsed_time = frame_end_time - frame_start_time;
233
234 running_time += frame_elapsed_time;
235
236 f64 remaining_seconds = target_frame_seconds - frame_elapsed_time;
237
238 if (remaining_seconds > 0) {
239 u64 remaining_ms = (remaining_seconds * 1000);
240
241 // If there is time left, give it back to the OS.
242 b8 limit_frames = FALSE;
243
244 if (remaining_ms > 0 && limit_frames) {
245 platform_sleep(remaining_ms - 1);
246 }
247
248 frame_count++;
249 }
250
251 // NOTE: Input update/state copying should always be handled
252 // after any input should be recorded; I.E. before this line.
253 // As a safety, input is the last thing to be updated before
254 // this frame ends.
255 input_update(delta);
256
257 // Update last time
258 app_state.last_time = current_time;
259 }
260 }
261
262 // Ensure running state is cleared
263 app_state.is_running = FALSE;
264
265 // Shutdown event system.
266 event_unregister(EVENT_CODE_APPLICATION_QUIT, 0, application_on_event);
267 event_unregister(EVENT_CODE_KEY_PRESSED, 0, application_on_key);
268 event_unregister(EVENT_CODE_KEY_RELEASED, 0, application_on_key);
269
270 event_shutdown();
271 input_shutdown();
272 renderer_shutdown();
273
274 // Clean up platform resources
275 platform_shutdown(&app_state.platform);
276
277 return TRUE;
278}

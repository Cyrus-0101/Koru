259b8 application_run() {
260 app_state->is_running = True;
261
262 clock_start(&app_state->clock);
263
264 clock_update(&app_state->clock);
265
266 app_state->last_time = app_state->clock.elapsed;
267
268 f64 running_time = 0;
269
270 u8 frame_count = 0;
271
272 f64 target_frame_seconds = 1.0f / 60;
273
274 // Log memory info - Memory Leak
275 KINFO(get_memory_usage_str());
276
277 while (app_state->is_running) {
278 if (!platform_pump_messages()) {
279 // If platform request to quit, stop running
280 app_state->is_running = False;
281 }
282
283 if (!app_state->is_suspended) {
284 // Update clock and get delta time.
285 clock_update(&app_state->clock);
286
287 f64 current_time = app_state->clock.elapsed;
288
289 f64 delta = (current_time - app_state->last_time);
290
291 f64 frame_start_time = platform_get_absolute_time();
292
293 if (!app_state->game_inst->update(app_state->game_inst, (f32)delta)) {
294 KFATAL("Game update failed. Shutting down!");
295
296 app_state->is_running = False;
297
298 break;
299 }
300
301 // Call game render routine
302 if (!app_state->game_inst->render(app_state->game_inst, (f32)delta)) {
303 KFATAL("Game render failed. Shutting down!");
304
305 app_state->is_running = False;
306
307 break;
308 }
309
310 // TODO: refactor packet creation
311 render_packet packet;
312
313 packet.delta_time = delta;
314
315 renderer_draw_frame(&packet);
316
317 // Figure out how long the frame took and, if below
318 f64 frame_end_time = platform_get_absolute_time();
319
320 f64 frame_elapsed_time = frame_end_time - frame_start_time;
321
322 running_time += frame_elapsed_time;
323
324 f64 remaining_seconds = target_frame_seconds - frame_elapsed_time;
325
326 if (remaining_seconds > 0) {
327 u64 remaining_ms = (remaining_seconds * 1000);
328
329 // If there is time left, give it back to the OS.
330 b8 limit_frames = False;
331
332 if (remaining_ms > 0 && limit_frames) {
333 platform_sleep(remaining_ms - 1);
334 }
335
336 frame_count++;
337 }
338
339 // NOTE: Input update/state copying should always be handled
340 // after any input should be recorded; I.E. before this line.
341 // As a safety, input is the last thing to be updated before
342 // this frame ends.
343 input_update(delta);
344
345 // Update last time
346 app_state->last_time = current_time;
347 }
348 }
349
350 // Ensure running state is cleared
351 app_state->is_running = False;
352
353 // Shutdown event system.
354 event_unregister(EVENT_CODE_APPLICATION_QUIT, 0, application_on_event);
355 event_unregister(EVENT_CODE_KEY_PRESSED, 0, application_on_key);
356 event_unregister(EVENT_CODE_KEY_RELEASED, 0, application_on_key);
357 event_unregister(EVENT_CODE_RESIZED, 0, application_on_resized);
358
359 input_system_shutdown(app_state->input_system_state);
360 renderer_system_shutdown(app_state->renderer_system_state);
361
362 // Clean up platform resources
363 platform_system_shutdown(app_state->platform_system_state);
364
365 shutdown_memory(app_state->memory_system_state);
366
367 event_system_shutdown(app_state->event_system_state);
368
369 return True;
370}

389b8 application_on_key(u16 code, void* sender, void* listener_inst, event_context context) {
390 if (code == EVENT_CODE_KEY_PRESSED) {
391 u16 key_code = context.data.u16[0];
392 if (key_code == KEY_ESCAPE) {
393 // NOTE: Technically firing an event to itself, but there may be other listeners.
394 event_context data = {};
395 event_fire(EVENT_CODE_APPLICATION_QUIT, 0, data);
396
397 // Block anything else from processing this.
398 return True;
399 } else if (key_code == KEY_A) {
400 // Example on checking for a key
401 KDEBUG("Explicit - A key pressed!");
402 } else {
403 KDEBUG("'%c' key pressed in window.", key_code);
404 }
405 } else if (code == EVENT_CODE_KEY_RELEASED) {
406 u16 key_code = context.data.u16[0];
407 if (key_code == KEY_B) {
408 // Example on checking for a key

411 KDEBUG("'%c' key released in window.", key_code);
412 }
413 }
414 return False;

417b8 application_on_resized(u16 code, void* sender, void* listener_inst, event_context context) {
418 if (code == EVENT_CODE_RESIZED) {
419 u16 width = context.data.u16[0];
420 u16 height = context.data.u16[1];
421
422 if (width != app_state->width || height != app_state->height) {
423 app_state->width = width;
424 app_state->height = height;
425
426 KDEBUG("Window resize: %i, %i", width, height);
427
428 // Handle minimization
429 if (width == 0 || height == 0) {
430 KINFO("Window minimized, suspending application.");
431 app_state->is_suspended = True;
432 return True;
433 } else {
434 if (app_state->is_suspended) {
435 KINFO("Window restored, resuming application.");
436 app_state->is_suspended = False;
437 }
438 app_state->game_inst->on_resize(app_state->game_inst, width, height);
439 renderer_on_resized(width, height);
440 }
441 }
442 }
443 return False;
444}

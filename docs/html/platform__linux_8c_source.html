60 // Create the internal state. Allocate memory for internal_state using malloc
61 plat_state->internal_state = malloc(sizeof(internal_state));
62
63 // Cast the void pointer to a typed pointer
64 internal_state* state = (internal_state*)plat_state->internal_state;
65
66 // Open a connection to the X11 server
67 state->display = XOpenDisplay(NULL); // NULL means default display
68
69 // Disable keyboard auto-repeat globally (games usually want discrete key presses)
70 XAutoRepeatOff(state->display);
71
72 // Get the modern XCB connection from the old Xlib Display
73 state->connection = XGetXCBConnection(state->display);
74
75 // Check if there was any error connecting
76 if (xcb_connection_has_error(state->connection)) {
77 KFATAL("Failed to connect to X server via XCB.");
78 return FALSE;
79 }
80
81 // Retrieve the setup data from the connection, akin to getting configuration
82 const struct xcb_setup_t* setup = xcb_get_setup(state->connection);
83
84 // Iterate through available screens (monitors) using an iterator
85 xcb_screen_iterator_t it = xcb_setup_roots_iterator(setup);
86
87 // We're choosing the first screen
88 int screen_p = 0;
89
90 for (i32 s = screen_p; s > 0; s--) {
91 xcb_screen_next(&it); // Move iterator forward
92 }
93
94 // Assign the selected screen
95 state->screen = it.data;
96
97 // Generate a unique XID for the window created
98 state->window = xcb_generate_id(state->connection);
99
100 // Register event types.
101 // XCB_CW_BACK_PIXEL = filling then window bg with a single colour
102 // XCB_CW_EVENT_MASK is required.
103 u32 event_mask = XCB_CW_BACK_PIXEL | XCB_CW_EVENT_MASK;
104
105 // Listen for keyboard and mouse buttons events
106 u32 event_values = XCB_EVENT_MASK_BUTTON_PRESS | XCB_EVENT_MASK_BUTTON_RELEASE |
107 XCB_EVENT_MASK_KEY_PRESS | XCB_EVENT_MASK_KEY_RELEASE |
108 XCB_EVENT_MASK_EXPOSURE | XCB_EVENT_MASK_POINTER_MOTION |
109 XCB_EVENT_MASK_STRUCTURE_NOTIFY;
110
111 // Values to be sent over XCB (bg colour, events)
112 u32 value_list[] = {state->screen->black_pixel, event_values};
113
117 xcb_void_cookie_t cookie = xcb_create_window(
118 state->connection,
119 XCB_COPY_FROM_PARENT, // Use same depth as parent
120 state->window, // Parent window (root window of screen)
121 state->screen->root, // parent
122 x, // x position
123 y, // y position
124 width, // width
125 height, // height
126 0, // No border
127 XCB_WINDOW_CLASS_INPUT_OUTPUT, // Can both draw and receive input
128 state->screen->root_visual, // Same color format as root window
129 event_mask, // Mask for attributes
130 value_list); // Values for those attributes
131
132 // Set the window title
133 xcb_change_property(
134 state->connection,
135 XCB_PROP_MODE_REPLACE, // Replace existing property
136 state->window,
137 XCB_ATOM_WM_NAME, // Property name (window title)
138 XCB_ATOM_STRING, // Type of data
139 8, // Format: data should be viewed 8 bits at a time
140 strlen(application_name), // Length of string
141 application_name); // The actual title string
142
147 xcb_intern_atom_cookie_t wm_delete_cookie = xcb_intern_atom(
148 state->connection,
149 0,
150 strlen("WM_DELETE_WINDOW"),
151 "WM_DELETE_WINDOW");
152 xcb_intern_atom_cookie_t wm_protocols_cookie = xcb_intern_atom(
153 state->connection,
154 0,
155 strlen("WM_PROTOCOLS"),
156 "WM_PROTOCOLS");
157
158 // Wait for replies
159 xcb_intern_atom_reply_t* wm_delete_reply = xcb_intern_atom_reply(
160 state->connection,
161 wm_delete_cookie,
162 NULL);
163 xcb_intern_atom_reply_t* wm_protocols_reply = xcb_intern_atom_reply(
164 state->connection,
165 wm_protocols_cookie,
166 NULL);
167
168 // Save atoms to internal state
169 state->wm_delete_win = wm_delete_reply->atom;
170 state->wm_protocols = wm_protocols_reply->atom;
171
172 // Let the window manager know we support WM_DELETE_WINDOW protocol
173 xcb_change_property(
174 state->connection,
175 XCB_PROP_MODE_REPLACE,
176 state->window,
177 wm_protocols_reply->atom,
178 4, // Format
179 32, // 32-bit values
180 1, // One element
181 &wm_delete_reply->atom); // Pointer to atom
182
183 // Map the window to the screen, make the window visible
184 xcb_map_window(state->connection, state->window);
185
186 // Send all previous commands to the X server
187 // Flush the stream
188 i32 stream_result = xcb_flush(state->connection);
189 if (stream_result <= 0) {
190 KFATAL("An error occurred when flusing the stream: %d", stream_result);
191 return FALSE;
192 }
193
194 return TRUE;
195}

215b8 platform_pump_messages(platform_state* plat_state) {
216 // Simply cold-cast to the known type.
217 internal_state* state = (internal_state*)plat_state->internal_state;
218
219 xcb_generic_event_t* event;
220 xcb_client_message_event_t* cm;
221
222 b8 quit_flagged = FALSE;
223
224 // Poll for all pending events
225 while ((event = xcb_poll_for_event(state->connection))) {
226 // Handle only one event per loop iteration
227 if (!event) {
228 break;
229 }
230
231 // Handle Input events
232 switch (event->response_type & ~0x80) {
233 case XCB_KEY_PRESS:
234 case XCB_KEY_RELEASE: {
235 // TODO: Key presses and releases
236 } break;
237
238 case XCB_BUTTON_PRESS:
239 case XCB_BUTTON_RELEASE: {
240 // TODO: Mouse button presses and releases handling
241 } break;
242
243 case XCB_MOTION_NOTIFY:
244 // TODO: Mouse movement
245 break;
246
247 case XCB_CONFIGURE_NOTIFY: {
248 // TODO: Resizing
249 } break;
250
251 case XCB_CLIENT_MESSAGE: {
252 cm = (xcb_client_message_event_t*)event;
253
254 // Window close
255 // If the message is WM_DELETE_WINDOW, user wants to quit
256 if (cm->data.data32[0] == state->wm_delete_win) {
257 quit_flagged = TRUE;
258 }
259 } break;
260
261 default:
262 // Ignore unknown events
263 break;
264 }
265
266 free(event); // Free the event after processing
267 }

269 return !quit_flagged;
270}

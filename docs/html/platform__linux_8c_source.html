130b8 platform_startup(
131 platform_state* plat_state,
132 const char* application_name,
133 i32 x,
134 i32 y,
135 i32 width,
136 i32 height) {
137 // Create the internal state. Allocate memory for internal_state using malloc
138 plat_state->internal_state = malloc(sizeof(internal_state));
139
140 // Cast the void pointer to a typed pointer
141 internal_state* state = (internal_state*)plat_state->internal_state;
142
143 // Open a connection to the X11 server
144 state->display = XOpenDisplay(NULL); // NULL means default display
145
146 // Disable keyboard auto-repeat globally (games usually want discrete key presses)
147 XAutoRepeatOff(state->display);
148
149 // Get the modern XCB connection from the old Xlib Display
150 state->connection = XGetXCBConnection(state->display);
151
152 // Check if there was any error connecting
153 if (xcb_connection_has_error(state->connection)) {
154 KFATAL("Failed to connect to X server via XCB.");
155 return False;
156 }
157
158 // Retrieve the setup data from the connection, akin to getting configuration
159 const struct xcb_setup_t* setup = xcb_get_setup(state->connection);
160
161 // Iterate through available screens (monitors) using an iterator
162 xcb_screen_iterator_t it = xcb_setup_roots_iterator(setup);
163
164 // We're choosing the first screen
165 int screen_p = 0;
166
167 for (i32 s = screen_p; s > 0; s--) {
168 xcb_screen_next(&it); // Move iterator forward
169 }
170
171 // Assign the selected screen
172 state->screen = it.data;
173
174 // Generate a unique XID for the window created
175 state->window = xcb_generate_id(state->connection);
176
177 // Register event types.
178 // XCB_CW_BACK_PIXEL = filling then window bg with a single colour
179 // XCB_CW_EVENT_MASK is required.
180 u32 event_mask = XCB_CW_BACK_PIXEL | XCB_CW_EVENT_MASK;
181
182 // Listen for keyboard and mouse buttons events
183 u32 event_values = XCB_EVENT_MASK_BUTTON_PRESS | XCB_EVENT_MASK_BUTTON_RELEASE |
184 XCB_EVENT_MASK_KEY_PRESS | XCB_EVENT_MASK_KEY_RELEASE |
185 XCB_EVENT_MASK_EXPOSURE | XCB_EVENT_MASK_POINTER_MOTION |
186 XCB_EVENT_MASK_STRUCTURE_NOTIFY;
187
188 // Values to be sent over XCB (bg colour, events)
189 u32 value_list[] = {state->screen->black_pixel, event_values};
190
194 xcb_void_cookie_t cookie = xcb_create_window(
195 state->connection,
196 XCB_COPY_FROM_PARENT, // Use same depth as parent
197 state->window, // Parent window (root window of screen)
198 state->screen->root, // parent
199 x, // x position
200 y, // y position
201 width, // width
202 height, // height
203 0, // No border
204 XCB_WINDOW_CLASS_INPUT_OUTPUT, // Can both draw and receive input
205 state->screen->root_visual, // Same color format as root window
206 event_mask, // Mask for attributes
207 value_list); // Values for those attributes
208
209 // Set the window title
210 xcb_change_property(
211 state->connection,
212 XCB_PROP_MODE_REPLACE, // Replace existing property
213 state->window,
214 XCB_ATOM_WM_NAME, // Property name (window title)
215 XCB_ATOM_STRING, // Type of data
216 8, // Format: data should be viewed 8 bits at a time
217 strlen(application_name), // Length of string
218 application_name); // The actual title string
219
224 xcb_intern_atom_cookie_t wm_delete_cookie = xcb_intern_atom(
225 state->connection,
226 0,
227 strlen("WM_DELETE_WINDOW"),
228 "WM_DELETE_WINDOW");
229 xcb_intern_atom_cookie_t wm_protocols_cookie = xcb_intern_atom(
230 state->connection,
231 0,
232 strlen("WM_PROTOCOLS"),
233 "WM_PROTOCOLS");
234
235 // Wait for replies
236 xcb_intern_atom_reply_t* wm_delete_reply = xcb_intern_atom_reply(
237 state->connection,
238 wm_delete_cookie,
239 NULL);
240 xcb_intern_atom_reply_t* wm_protocols_reply = xcb_intern_atom_reply(
241 state->connection,
242 wm_protocols_cookie,
243 NULL);
244
245 // Save atoms to internal state
246 state->wm_delete_win = wm_delete_reply->atom;
247 state->wm_protocols = wm_protocols_reply->atom;
248
249 // Let the window manager know we support WM_DELETE_WINDOW protocol
250 xcb_change_property(
251 state->connection,
252 XCB_PROP_MODE_REPLACE,
253 state->window,
254 wm_protocols_reply->atom,
255 4, // Format
256 32, // 32-bit values
257 1, // One element
258 &wm_delete_reply->atom); // Pointer to atom
259
260 // Map the window to the screen, make the window visible
261 xcb_map_window(state->connection, state->window);
262
263 // Send all previous commands to the X server
264 // Flush the stream
265 i32 stream_result = xcb_flush(state->connection);
266 if (stream_result <= 0) {
267 KFATAL("An error occurred when flusing the stream: %d", stream_result);
268 return False;
269 }
270
271 state->is_visible = True;
272
273 return True;
274}

304b8 platform_pump_messages(platform_state* plat_state) {
305 // Simply cold-cast to the known type.
306 internal_state* state = (internal_state*)plat_state->internal_state;
307
308 xcb_generic_event_t* event;
309 xcb_client_message_event_t* cm;
310
311 b8 quit_flagged = False;
312
313 // Poll for all pending events
314 while ((event = xcb_poll_for_event(state->connection))) {
315 // Handle only one event per loop iteration
316 if (!event) {
317 break;
318 }
319
320 // Handle Input events
321 switch (event->response_type & ~0x80) {
322 case XCB_KEY_PRESS:
323 case XCB_KEY_RELEASE: {
324 // Key press event - xcb_key_press_event_t / xcb_key_release_event_t are similar
325 xcb_key_press_event_t* kb_event = (xcb_key_press_event_t*)event;
326 b8 pressed = event->response_type == XCB_KEY_PRESS;
327
328 xcb_keycode_t code = kb_event->detail;
329
330 KeySym key_sym = XkbKeycodeToKeysym(
331 state->display,
332 (KeyCode)code, // event.xkey.keycode
333 0,
334 code & ShiftMask ? 1 : 0);
335
336 keys key = translate_keycode(key_sym);
337
338 // Pass to the input subsystem for processing.
339 input_process_key(key, pressed);
340 } break;
341
342 case XCB_BUTTON_PRESS:
343 case XCB_BUTTON_RELEASE: {
344 xcb_button_press_event_t* mouse_event = (xcb_button_press_event_t*)event;
345
346 b8 pressed = event->response_type == XCB_BUTTON_PRESS;
347
348 buttons mouse_button = BUTTON_MAX_BUTTONS;
349
350 switch (mouse_event->detail) {
351 case XCB_BUTTON_INDEX_1:
352 mouse_button = BUTTON_LEFT;
353
354 break;
355 case XCB_BUTTON_INDEX_2:
356 mouse_button = BUTTON_MIDDLE;
357
358 break;
359 case XCB_BUTTON_INDEX_3:
360 mouse_button = BUTTON_RIGHT;
361
362 break;
363 }
364
365 // Pass over to the input subsystem.
366 if (mouse_button != BUTTON_MAX_BUTTONS) {
367 input_process_button(mouse_button, pressed);
368 }
369 } break;
370
371 case XCB_MOTION_NOTIFY: {
372 // Mouse move
373 xcb_motion_notify_event_t* move_event = (xcb_motion_notify_event_t*)event;
374
375 // Pass to the input subsystem.
376 input_process_mouse_move(move_event->event_x, move_event->event_y);
377 } break;
378
379 case XCB_CONFIGURE_NOTIFY: {
380 /*
381 * Resizing - note that this is also triggered by moving the window, but should be
382 * passed anyway since a change in the x/y could mean an upper-left resize.
383 * The application layer can decide what to do with this.
384 */
385 xcb_configure_notify_event_t* configure_event = (xcb_configure_notify_event_t*)event;
386
387 state->is_visible = (configure_event->width > 0 && configure_event->height > 0);
388
389 /*
390 * Fire the event. The application layer should pick this up,
391 * but not handle it as it shouldn't be visible to other parts
392 * of the application.
393 */
394 event_context context;
395
396 context.data.u16[0] = configure_event->width;
397 context.data.u16[1] = configure_event->height;
398 context.data.u16[2] = (u16)state->is_visible; // Pass visibility flag
399
400 event_fire(EVENT_CODE_RESIZED, 0, context);
401 } break;
402
403 case XCB_CLIENT_MESSAGE: {
404 cm = (xcb_client_message_event_t*)event;
405
406 // Window close
407 // If the message is WM_DELETE_WINDOW, user wants to quit
408 if (cm->data.data32[0] == state->wm_delete_win) {
409 quit_flagged = True;
410 }
411 } break;
412
413 default:
414 // Ignore unknown events
415 break;
416 }
417
418 free(event); // Free the event after processing
419 }
420
421 return !quit_flagged;
422}

72 // Create the internal state. Allocate memory for internal_state using malloc
73 plat_state->internal_state = malloc(sizeof(internal_state));
74
75 // Cast the void pointer to a typed pointer
76 internal_state* state = (internal_state*)plat_state->internal_state;
77
78 // Open a connection to the X11 server
79 state->display = XOpenDisplay(NULL); // NULL means default display
80
81 // Disable keyboard auto-repeat globally (games usually want discrete key presses)
82 XAutoRepeatOff(state->display);
83
84 // Get the modern XCB connection from the old Xlib Display
85 state->connection = XGetXCBConnection(state->display);
86
87 // Check if there was any error connecting
88 if (xcb_connection_has_error(state->connection)) {
89 KFATAL("Failed to connect to X server via XCB.");
90 return FALSE;
91 }
92
93 // Retrieve the setup data from the connection, akin to getting configuration
94 const struct xcb_setup_t* setup = xcb_get_setup(state->connection);
95
96 // Iterate through available screens (monitors) using an iterator
97 xcb_screen_iterator_t it = xcb_setup_roots_iterator(setup);
98
99 // We're choosing the first screen
100 int screen_p = 0;
101
102 for (i32 s = screen_p; s > 0; s--) {
103 xcb_screen_next(&it); // Move iterator forward
104 }
105
106 // Assign the selected screen
107 state->screen = it.data;
108
109 // Generate a unique XID for the window created
110 state->window = xcb_generate_id(state->connection);
111
112 // Register event types.
113 // XCB_CW_BACK_PIXEL = filling then window bg with a single colour
114 // XCB_CW_EVENT_MASK is required.
115 u32 event_mask = XCB_CW_BACK_PIXEL | XCB_CW_EVENT_MASK;
116
117 // Listen for keyboard and mouse buttons events
118 u32 event_values = XCB_EVENT_MASK_BUTTON_PRESS | XCB_EVENT_MASK_BUTTON_RELEASE |
119 XCB_EVENT_MASK_KEY_PRESS | XCB_EVENT_MASK_KEY_RELEASE |
120 XCB_EVENT_MASK_EXPOSURE | XCB_EVENT_MASK_POINTER_MOTION |
121 XCB_EVENT_MASK_STRUCTURE_NOTIFY;
122
123 // Values to be sent over XCB (bg colour, events)
124 u32 value_list[] = {state->screen->black_pixel, event_values};
125
129 xcb_void_cookie_t cookie = xcb_create_window(
130 state->connection,
131 XCB_COPY_FROM_PARENT, // Use same depth as parent
132 state->window, // Parent window (root window of screen)
133 state->screen->root, // parent
134 x, // x position
135 y, // y position
136 width, // width
137 height, // height
138 0, // No border
139 XCB_WINDOW_CLASS_INPUT_OUTPUT, // Can both draw and receive input
140 state->screen->root_visual, // Same color format as root window
141 event_mask, // Mask for attributes
142 value_list); // Values for those attributes
143
144 // Set the window title
145 xcb_change_property(
146 state->connection,
147 XCB_PROP_MODE_REPLACE, // Replace existing property
148 state->window,
149 XCB_ATOM_WM_NAME, // Property name (window title)
150 XCB_ATOM_STRING, // Type of data
151 8, // Format: data should be viewed 8 bits at a time
152 strlen(application_name), // Length of string
153 application_name); // The actual title string
154
159 xcb_intern_atom_cookie_t wm_delete_cookie = xcb_intern_atom(
160 state->connection,
161 0,
162 strlen("WM_DELETE_WINDOW"),
163 "WM_DELETE_WINDOW");
164 xcb_intern_atom_cookie_t wm_protocols_cookie = xcb_intern_atom(
165 state->connection,
166 0,
167 strlen("WM_PROTOCOLS"),
168 "WM_PROTOCOLS");
169
170 // Wait for replies
171 xcb_intern_atom_reply_t* wm_delete_reply = xcb_intern_atom_reply(
172 state->connection,
173 wm_delete_cookie,
174 NULL);
175 xcb_intern_atom_reply_t* wm_protocols_reply = xcb_intern_atom_reply(
176 state->connection,
177 wm_protocols_cookie,
178 NULL);
179
180 // Save atoms to internal state
181 state->wm_delete_win = wm_delete_reply->atom;
182 state->wm_protocols = wm_protocols_reply->atom;
183
184 // Let the window manager know we support WM_DELETE_WINDOW protocol
185 xcb_change_property(
186 state->connection,
187 XCB_PROP_MODE_REPLACE,
188 state->window,
189 wm_protocols_reply->atom,
190 4, // Format
191 32, // 32-bit values
192 1, // One element
193 &wm_delete_reply->atom); // Pointer to atom
194
195 // Map the window to the screen, make the window visible
196 xcb_map_window(state->connection, state->window);
197
198 // Send all previous commands to the X server
199 // Flush the stream
200 i32 stream_result = xcb_flush(state->connection);
201 if (stream_result <= 0) {
202 KFATAL("An error occurred when flusing the stream: %d", stream_result);
203 return FALSE;
204 }
205
206 return TRUE;
207}

227b8 platform_pump_messages(platform_state* plat_state) {
228 // Simply cold-cast to the known type.
229 internal_state* state = (internal_state*)plat_state->internal_state;
230
231 xcb_generic_event_t* event;
232 xcb_client_message_event_t* cm;
233
234 b8 quit_flagged = FALSE;
235
236 // Poll for all pending events
237 while ((event = xcb_poll_for_event(state->connection))) {
238 // Handle only one event per loop iteration
239 if (!event) {
240 break;
241 }
242
243 // Handle Input events
244 switch (event->response_type & ~0x80) {
245 case XCB_KEY_PRESS:
246 case XCB_KEY_RELEASE: {
247 // TODO: Key presses and releases
248 } break;
249
250 case XCB_BUTTON_PRESS:
251 case XCB_BUTTON_RELEASE: {
252 // TODO: Mouse button presses and releases handling
253 } break;
254
255 case XCB_MOTION_NOTIFY:
256 // TODO: Mouse movement
257 break;
258
259 case XCB_CONFIGURE_NOTIFY: {
260 // TODO: Resizing
261 } break;
262
263 case XCB_CLIENT_MESSAGE: {
264 cm = (xcb_client_message_event_t*)event;
265
266 // Window close
267 // If the message is WM_DELETE_WINDOW, user wants to quit
268 if (cm->data.data32[0] == state->wm_delete_win) {
269 quit_flagged = TRUE;
270 }
271 } break;
272
273 default:
274 // Ignore unknown events
275 break;
276 }
277
278 free(event); // Free the event after processing
279 }
280
281 return !quit_flagged;
282}

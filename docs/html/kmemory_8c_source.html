211 char buffer[8000] = "System memory use (tagged):\n";

218
219 if (state_ptr->stats.tagged_allocations[i] >= gib) {
220 unit[0] = 'G';
221 amount = state_ptr->stats.tagged_allocations[i] / (float)gib;
222 } else if (state_ptr->stats.tagged_allocations[i] >= mib) {

225 } else if (state_ptr->stats.tagged_allocations[i] >= kib) {

228 } else {

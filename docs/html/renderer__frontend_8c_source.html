147b8 renderer_draw_frame(render_packet* packet) {
148 // Begin the frame
149 if (renderer_begin_frame(packet->delta_time)) {
150 state_ptr->backend.update_global_state(state_ptr->projection, state_ptr->view, vec3_zero(), vec4_one(), 0);
151
152 // mat4 model = mat4_translation((vec3){0, 0, 0});
153 static f32 angle = 0.01f;
154 angle += 0.001f;
155 quat rotation = quat_from_axis_angle(vec3_forward(), angle, False);
156 mat4 model = quat_to_rotation_matrix(rotation, vec3_zero());
157
158 state_ptr->backend.update_object(model);
159
160 // End the frame. If this fails, it is likely unrecoverable.
161 b8 result = renderer_end_frame(packet->delta_time);
162
163 if (!result) {
164 KERROR("renderer_end_frame failed. Application shutting down...");
165 return False;
166 }
167 }

169 return True;

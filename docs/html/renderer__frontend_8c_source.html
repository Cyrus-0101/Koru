105b8 load_texture(const char* texture_name, texture* t) {
106 // TO-DO: Should be able to be located anywhere
107 char* format_st = "assets/textures/%s.%s"; // File_name.extension
108 const i32 required_channel_count = 4;
109 stbi_set_flip_vertically_on_load(True);
110 char full_file_path[512];
111
112 // TO-DO: Try different extensions (compatible)
113 string_format(full_file_path, format_st, texture_name, "png");
114
115 // Use a temporary local texture to load state
116 texture temp_texture;
117
118 // Open the file and load the data
119 u8* data = stbi_load(
120 full_file_path,
121 (i32*)&temp_texture.width,
122 (i32*)&temp_texture.height,
123 (i32*)&temp_texture.channel_count,
124 required_channel_count);
125
126 // Overwrite channel count to required_channel_count
127 temp_texture.channel_count = required_channel_count;
128
129 if (data) {
130 KINFO("Loaded texture: %s (%i x %i x %i)", full_file_path, temp_texture.width, temp_texture.height, temp_texture.channel_count);
131
132 u32 current_generation = t->generation;
133 t->generation = INVALID_ID; // Mark as invalid while updating
134
135 u64 total_size = temp_texture.width * temp_texture.height * temp_texture.channel_count;
136 // Check for transparency
137 b32 has_transparency = False;
138 for (u64 i = 0; i < total_size; i += required_channel_count) {
139 if (data[i + 3] < 255) {
140 has_transparency = True;
141 break;
142 }
143 }
144
145 // Returns 0 on failure
146 if (stbi_failure_reason()) {
147 KWARN("Failed to load texture %s: %s", full_file_path, stbi_failure_reason());
148 }
149
150 // Acquire internal texture data and upload to GPU
151 // TO-DO: Make auto_release configurable
152 renderer_create_texture(
153 texture_name,
154 True,
155 temp_texture.width,
156 temp_texture.height,
157 temp_texture.channel_count,
158 data,
159 has_transparency,
160 &temp_texture);
161
162 // Take a copy of the old texture
163 texture old = *t;
164
165 // Assign the temp texture to the pointer
166 *t = temp_texture;
167
168 // Destroy the old texture if valid
169 renderer_destroy_texture(&old);
170
171 if (current_generation == INVALID_ID) {
172 t->generation = 0;
173 } else {
174 t->generation = current_generation + 1;
175 }
176
177 // CLean up data
178 stbi_image_free(data);
179
180 return True;
181 } else {
182 if (stbi_failure_reason()) {
183 KWARN("Failed to load texture %s: %s", full_file_path, stbi_failure_reason());
184 }
185
186 return False;
187 }
188}

219b8 renderer_system_initialize(u64* memory_requirement, void* state, const char* application_name) {
220 *memory_requirement = sizeof(renderer_system_state);
221 if (state == 0) {
222 return True; // No state provided, nothing to initialize
223 }
224
225 state_ptr = state;
226
227 // TO-DO: Temporary Code
228 event_register(EVENT_CODE_DEBUG0, state_ptr, event_on_debug_event);
229 // TO-DO: End of Temporary Code
230
231 // Take a pointer to default textures for use in the backend
232 state_ptr->backend.default_diffuse = &state_ptr->default_texture;
233
234 // TODO: Make backend type configurable via config or runtime settings
235 renderer_backend_create(RENDERER_BACKEND_TYPE_VULKAN, &state_ptr->backend);
236 // Initialize the frame counter
237 state_ptr->backend.frame_number = 0;
238
239 // Call the backend's initialization routine
240 if (!state_ptr->backend.initialize(&state_ptr->backend, application_name)) {
241 KFATAL("Renderer backend failed to initialize. Shutting down.");
242 return False;
243 }
244
245 state_ptr->near_clip = 0.1f;
246 state_ptr->far_clip = 1000.0f;
247 state_ptr->projection = mat4_perspective(deg_to_rad(45.0f), 1280 / 720.0f, state_ptr->near_clip, state_ptr->far_clip);
248
249 state_ptr->view = mat4_translation((vec3){0, 0, -30.0f}); //-30.0f
250 state_ptr->view = mat4_inverse(state_ptr->view);
251
252 // NOTE: Create default texture, a 256x256 blue/white checkerboard pattern.
253 // This is done in code to eliminate asset dependencies.
254 KTRACE("Creating default texture...");
255
256 u8 pixels[PIXEL_COUNT * CHANNELS];
257 // u8* pixels = kallocate(sizeof(u8) * pixel_count * bpp, MEMORY_TAG_TEXTURE);
258
259 kset_memory(pixels, 255, sizeof(u8) * PIXEL_COUNT * CHANNELS);
260
261 // Each pixel.
262 for (u64 row = 0; row < TEX_DIMENSIONS; ++row) {
263 for (u64 col = 0; col < TEX_DIMENSIONS; ++col) {
264 u64 index = (row * TEX_DIMENSIONS) + col;
265 u64 index_bpp = index * CHANNELS;
266
267 if (row % 2) {
268 if (col % 2) {
269 pixels[index_bpp + 0] = 0;
270 pixels[index_bpp + 1] = 0;
271 }
272 } else {
273 if (!(col % 2)) {
274 pixels[index_bpp + 0] = 0;
275 pixels[index_bpp + 1] = 0;
276 }
277 }
278 }
279 }

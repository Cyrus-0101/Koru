154b8 renderer_draw_frame(render_packet* packet) {
155 // Begin the frame
156 if (renderer_begin_frame(packet->delta_time)) {
157 state_ptr->backend.update_global_state(state_ptr->projection, state_ptr->view, vec3_zero(), vec4_one(), 0);
158
159 u32 count = packet->geometry_count;
160 for (u32 i = 0; i < count; ++i) {
161 state_ptr->backend.draw_geometry(packet->geometries[i]);

163
164 // End the frame. If this fails, it is likely unrecoverable.

169 return False;
170 }
171 }

173 return True;

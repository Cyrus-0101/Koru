556b8 vulkan_renderer_backend_end_frame(renderer_backend* backend, f32 delta_time) {
557 vulkan_command_buffer* command_buffer = &context.graphics_command_buffers[context.image_index];
558
559 // End renderpass
560 vulkan_renderpass_end(command_buffer, &context.main_renderpass);
561
562 vulkan_command_buffer_end(command_buffer);
563
564 // Make sure the previous frame is not using this image (i.e. its fence is being waited on)
565 if (context.images_in_flight[context.image_index] != VK_NULL_HANDLE) { // was frame
566 vulkan_fence_wait(
567 &context,
568 context.images_in_flight[context.image_index],
569 UINT64_MAX);
570 }
571
572 // Mark the image fence as in-use by this frame.
573 context.images_in_flight[context.image_index] = &context.in_flight_fences[context.current_frame];
574
575 // Reset the fence for use on the next frame
576 vulkan_fence_reset(&context, &context.in_flight_fences[context.current_frame]);
577
578 // Submit the queue and wait for the operation to complete.
579 // Begin queue submission
580 VkSubmitInfo submit_info = {VK_STRUCTURE_TYPE_SUBMIT_INFO};
581
582 // Command buffer(s) to be executed.
583 submit_info.commandBufferCount = 1;
584 submit_info.pCommandBuffers = &command_buffer->handle;
585
586 // The semaphore(s) to be signaled when the queue is complete.
587 submit_info.signalSemaphoreCount = 1;
588 submit_info.pSignalSemaphores = &context.queue_complete_semaphores[context.current_frame];
589
590 // Wait semaphore ensures that the operation cannot begin until the image is available.
591 submit_info.waitSemaphoreCount = 1;
592 submit_info.pWaitSemaphores = &context.image_available_semaphores[context.current_frame];
593
594 /*
595 * Each semaphore waits on the corresponding pipeline stage to complete. 1:1 ratio.
596 * VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT prevents subsequent colour attachment
597 * writes from executing until the semaphore signals (i.e. one frame is presented at a time)
598 */
599 VkPipelineStageFlags flags[1] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
600 submit_info.pWaitDstStageMask = flags;
601
602 VkResult result = vkQueueSubmit(
603 context.device.graphics_queue,
604 1,
605 &submit_info,
606 context.in_flight_fences[context.current_frame].handle);
607
608 if (result != VK_SUCCESS) {
609 KERROR("vkQueueSubmit failed with result: %s", vulkan_result_string(result, True));
610
611 return False;
612 }
613
614 vulkan_command_buffer_update_submitted(command_buffer); // End queue submission
615
616 // Give the image back to the swapchain.
617 vulkan_swapchain_present(
618 &context,
619 &context.swapchain,
620 context.device.graphics_queue,
621 context.device.present_queue,
622 context.queue_complete_semaphores[context.current_frame],
623 context.image_index);
624
625 return True;
626}

693i32 find_memory_index(u32 type_filter, u32 property_flags) {
694 KDEBUG("Finding memory index for filter %u, flags %u...", type_filter, property_flags);
695
696 VkPhysicalDeviceMemoryProperties memory_properties;
697 vkGetPhysicalDeviceMemoryProperties(context.device.physical_device, &memory_properties);
698
699 for (u32 i = 0; i < memory_properties.memoryTypeCount; ++i) {
700 // Check each memory type to see if its bit is set to 1.
701 if (type_filter & (1 << i) && (memory_properties.memoryTypes[i].propertyFlags & property_flags) == property_flags) {
702 KDEBUG("Found valid memory index: %u", i);
703 return i;
704 }
705 }
706
707 KWARN("Unable to find suitable memory type!");
708 return -1;
709}

758b8 recreate_swapchain(renderer_backend* backend) {
759 KINFO("Recreating");
760 // If already being recreated, do not try again.
761 if (context.recreating_swapchain) {
762 KDEBUG("recreate_swapchain called when already recreating, skipping!");
763 return False;
764 }
765
766 // Detect if the window is too small to be drawn to
767 if (context.framebuffer_width == 0 || context.framebuffer_height == 0) {
768 KDEBUG("recreate_swapchain called when window is < 1 in a dimension. Booting.");
769 return False;
770 }
771
772 // Mark as recreating if the dimensions are valid.
773 context.recreating_swapchain = True;
774
775 // Wait for any operations to complete.
776 vkDeviceWaitIdle(context.device.logical_device);
777
778 // Clear these out just in case.
779 for (u32 i = 0; i < context.swapchain.image_count; ++i) {
780 context.images_in_flight[i] = 0;
781 }
782
783 // Requery support
784 vulkan_device_query_swapchain_support(
785 context.device.physical_device,
786 context.surface,
787 &context.device.swapchain_support);
788
789 vulkan_device_detect_depth_format(&context.device);
790
791 vulkan_swapchain_recreate(
792 &context,
793 cached_framebuffer_width,
794 cached_framebuffer_height,
795 &context.swapchain);
796
797 // Sync the framebuffer size with the cached sizes.
798 context.framebuffer_width = cached_framebuffer_width;
799 context.framebuffer_height = cached_framebuffer_height;
800 context.main_renderpass.w = context.framebuffer_width;
801 context.main_renderpass.h = context.framebuffer_height;
802 cached_framebuffer_width = 0;
803 cached_framebuffer_height = 0;
804
805 // Update framebuffer size generation.
806 context.framebuffer_size_last_generation = context.framebuffer_size_generation;
807
808 // Cleanup swapchain
809 for (u32 i = 0; i < context.swapchain.image_count; ++i) {
810 vulkan_command_buffer_free(&context, context.device.graphics_command_pool, &context.graphics_command_buffers[i]);
811 }
812
813 // Framebuffers.
814 for (u32 i = 0; i < context.swapchain.image_count; ++i) {
815 vulkan_framebuffer_destroy(&context, &context.swapchain.framebuffers[i]);
816 }
817
818 context.main_renderpass.x = 0;
819 context.main_renderpass.y = 0;
820 context.main_renderpass.w = context.framebuffer_width;
821 context.main_renderpass.h = context.framebuffer_height;
822
823 regenerate_framebuffers(backend, &context.swapchain, &context.main_renderpass);
824
825 create_command_buffers(backend);
826
827 // Clear the recreating flag.
828 context.recreating_swapchain = False;
829
830 return True;
831}

868void vulkan_renderer_create_texture(const char* name, i32 width, i32 height, i32 channel_count, const u8* pixels, b8 has_transparency, texture* out_texture) {
869 out_texture->width = width;
870 out_texture->height = height;
871 out_texture->channel_count = channel_count;
872 out_texture->generation = INVALID_ID;
873
874 // Internal data creation.
875 // TODO: Use an allocator for this.
876 out_texture->internal_data = (vulkan_texture_data*)kallocate(sizeof(vulkan_texture_data), MEMORY_TAG_TEXTURE);
877 vulkan_texture_data* data = (vulkan_texture_data*)out_texture->internal_data;
878
879 VkDeviceSize image_size = width * height * channel_count;
880
881 // NOTE: Assumes 8 bits per channel.
882 VkFormat image_format = VK_FORMAT_R8G8B8A8_UNORM;
883
884 // Create a staging buffer and load data into it.
885 VkBufferUsageFlags usage = VK_BUFFER_USAGE_TRANSFER_SRC_BIT;
886
887 VkMemoryPropertyFlags memory_prop_flags = VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT;
888
889 vulkan_buffer staging;
890
891 vulkan_buffer_create(&context, image_size, usage, memory_prop_flags, True, &staging);
892
893 vulkan_buffer_load_data(&context, &staging, 0, image_size, 0, pixels);
894
895 // NOTE: Lots of assumptions here, different texture types will require
896 // different options here.
897 vulkan_image_create(
898 &context,
899 VK_IMAGE_TYPE_2D,
900 width,
901 height,
902 image_format,
903 VK_IMAGE_TILING_OPTIMAL,
904 VK_IMAGE_USAGE_TRANSFER_SRC_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT | VK_IMAGE_USAGE_SAMPLED_BIT | VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT,
905 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT,
906 True,
907 VK_IMAGE_ASPECT_COLOR_BIT,
908 &data->image);
909
910 vulkan_command_buffer temp_buffer;
911
912 VkCommandPool pool = context.device.graphics_command_pool;
913
914 VkQueue queue = context.device.graphics_queue;
915
916 vulkan_command_buffer_allocate_and_begin_single_use(&context, pool, &temp_buffer);
917
918 // Transition the layout from whatever it is currently to optimal for recieving data.
919 vulkan_image_transition_layout(
920 &context,
921 &temp_buffer,
922 &data->image,
923 image_format,
924 VK_IMAGE_LAYOUT_UNDEFINED,
925 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL);
926
927 // Copy the data from the buffer.
928 vulkan_image_copy_from_buffer(&context, &data->image, staging.handle, &temp_buffer);
929
930 // Transition from optimal for data reciept to shader-read-only optimal layout.
931 vulkan_image_transition_layout(
932 &context,
933 &temp_buffer,
934 &data->image,
935 image_format,
936 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
937 VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
938
939 // End the command buffer and wait for execution (ensures copy is done)
940 vulkan_command_buffer_end_single_use(&context, pool, &temp_buffer, queue);
941
942 // Destroy buffer
943 vulkan_buffer_destroy(&context, &staging);
944
945 // Create a sampler for the texture
946 VkSamplerCreateInfo sampler_info = {VK_STRUCTURE_TYPE_SAMPLER_CREATE_INFO};
947
948 // TODO: These filters should be configurable.
949 sampler_info.magFilter = VK_FILTER_LINEAR;
950 sampler_info.minFilter = VK_FILTER_LINEAR;
951
952 sampler_info.addressModeU = VK_SAMPLER_ADDRESS_MODE_REPEAT;
953 sampler_info.addressModeV = VK_SAMPLER_ADDRESS_MODE_REPEAT;
954 sampler_info.addressModeW = VK_SAMPLER_ADDRESS_MODE_REPEAT;
955
956 sampler_info.anisotropyEnable = VK_TRUE;
957 sampler_info.maxAnisotropy = 16;
958
959 sampler_info.borderColor = VK_BORDER_COLOR_INT_OPAQUE_BLACK;
960 sampler_info.unnormalizedCoordinates = VK_FALSE;
961
962 sampler_info.compareEnable = VK_FALSE;
963 sampler_info.compareOp = VK_COMPARE_OP_ALWAYS;
964
965 sampler_info.mipmapMode = VK_SAMPLER_MIPMAP_MODE_LINEAR;
966 sampler_info.mipLodBias = 0.0f;
967
968 sampler_info.minLod = 0.0f;
969 sampler_info.maxLod = 0.0f;
970
971 VkResult result = vkCreateSampler(context.device.logical_device, &sampler_info, context.allocator, &data->sampler);
972
973 if (!vulkan_result_is_success(VK_SUCCESS)) {
974 KERROR("Error creating texture sampler: %s", vulkan_result_string(result, True));
975
976 return;
977 }
978
979 out_texture->has_transparency = has_transparency;
980 out_texture->generation++;
981}

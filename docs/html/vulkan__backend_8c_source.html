519 vulkan_command_buffer* command_buffer = &context.graphics_command_buffers[context.image_index];
520
521 // End renderpass
522 vulkan_renderpass_end(command_buffer, &context.main_renderpass);
523
524 vulkan_command_buffer_end(command_buffer);
525
526 // Make sure the previous frame is not using this image (i.e. its fence is being waited on)
527 if (context.images_in_flight[context.image_index] != VK_NULL_HANDLE) { // was frame
528 vulkan_fence_wait(
529 &context,
530 context.images_in_flight[context.image_index],
531 UINT64_MAX);
532 }
533
534 // Mark the image fence as in-use by this frame.
535 context.images_in_flight[context.image_index] = &context.in_flight_fences[context.current_frame];
536
537 // Reset the fence for use on the next frame
538 vulkan_fence_reset(&context, &context.in_flight_fences[context.current_frame]);
539
540 // Submit the queue and wait for the operation to complete.
541 // Begin queue submission
542 VkSubmitInfo submit_info = {VK_STRUCTURE_TYPE_SUBMIT_INFO};
543
544 // Command buffer(s) to be executed.
545 submit_info.commandBufferCount = 1;
546 submit_info.pCommandBuffers = &command_buffer->handle;
547
548 // The semaphore(s) to be signaled when the queue is complete.
549 submit_info.signalSemaphoreCount = 1;
550 submit_info.pSignalSemaphores = &context.queue_complete_semaphores[context.current_frame];
551

553 submit_info.waitSemaphoreCount = 1;

555
556 /*
557 * Each semaphore waits on the corresponding pipeline stage to complete. 1:1 ratio.
558 * VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT prevents subsequent colour attachment
559 * writes from executing until the semaphore signals (i.e. one frame is presented at a time)
560 */
561 VkPipelineStageFlags flags[1] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
562 submit_info.pWaitDstStageMask = flags;
563
564 VkResult result = vkQueueSubmit(
565 context.device.graphics_queue,
566 1,
567 &submit_info,
568 context.in_flight_fences[context.current_frame].handle);
569
570 if (result != VK_SUCCESS) {
571 KERROR("vkQueueSubmit failed with result: %s", vulkan_result_string(result, True));
572
573 return False;
574 }
575
576 vulkan_command_buffer_update_submitted(command_buffer); // End queue submission
577
578 // Give the image back to the swapchain.
579 vulkan_swapchain_present(
580 &context,
581 &context.swapchain,
582 context.device.graphics_queue,
583 context.device.present_queue,
584 context.queue_complete_semaphores[context.current_frame],
585 context.image_index);

679 KINFO("Recreating");
680 // If already being recreated, do not try again.
681 if (context.recreating_swapchain) {
682 KDEBUG("recreate_swapchain called when already recreating, skipping!");

408b8 vulkan_renderer_backend_begin_frame(renderer_backend* backend, f32 delta_time) {
409 vulkan_device* device = &context.device;
410
411 // Check if recreating swap chain and boot out.
412 if (context.recreating_swapchain) {
413 VkResult result = vkDeviceWaitIdle(device->logical_device);
414
415 if (!vulkan_result_is_success(result)) {
416 KERROR("vulkan_renderer_backend_begin_frame vkDeviceWaitIdle (1) failed: '%s'", vulkan_result_string(result, TRUE));
417 return FALSE;
418 }
419
420 KINFO("Recreating swapchain, booting.");
421 return FALSE;
422 }
423
424 // Check if the framebuffer has been resized. If so, a new swapchain must be created.
425 if (context.framebuffer_size_generation != context.framebuffer_size_last_generation) {
426 VkResult result = vkDeviceWaitIdle(device->logical_device);
427 if (!vulkan_result_is_success(result)) {
428 KERROR("vulkan_renderer_backend_begin_frame vkDeviceWaitIdle (2) failed: '%s'", vulkan_result_string(result, TRUE));
429 return FALSE;
430 }
431
432 // If the swapchain recreation failed (because, for example, the window was minimized),
433 // boot out before unsetting the flag.
434 if (!recreate_swapchain(backend)) {
435 return FALSE;
436 }
437
438 KINFO("Resized, booting.");
439 return FALSE;
440 }
441
442 // Wait for the execution of the current frame to complete. The fence being free will allow this one to move on.
443 if (!vulkan_fence_wait(
444 &context,
445 &context.in_flight_fences[context.current_frame],
446 UINT64_MAX)) {
447 KWARN("In-flight fence wait failure!");
448
449 return FALSE;
450 }
451
452 // Acquire the next image from the swap chain. Pass along the semaphore that should signaled when this completes.
453 // This same semaphore will later be waited on by the queue submission to ensure this image is available.
454 if (!vulkan_swapchain_acquire_next_image_index(
455 &context,
456 &context.swapchain,
457 UINT64_MAX,
458 context.image_available_semaphores[context.current_frame],
459 0,
460 &context.image_index)) {
461 return FALSE;
462 }
463
464 // Begin recording commands.
465 vulkan_command_buffer* command_buffer = &context.graphics_command_buffers[context.image_index];
466 vulkan_command_buffer_reset(command_buffer);
467 vulkan_command_buffer_begin(command_buffer, FALSE, FALSE, FALSE);
468
469 // Dynamic state
470 VkViewport viewport;
471 viewport.x = 0.0f;
472 viewport.y = (f32)context.framebuffer_height;
473 viewport.width = (f32)context.framebuffer_width;
474 viewport.height = -(f32)context.framebuffer_height;
475 viewport.minDepth = 0.0f;
476 viewport.maxDepth = 1.0f;
477
478 // Scissor
479 VkRect2D scissor;
480 scissor.offset.x = scissor.offset.y = 0;
481 scissor.extent.width = context.framebuffer_width;
482 scissor.extent.height = context.framebuffer_height;
483
484 vkCmdSetViewport(command_buffer->handle, 0, 1, &viewport);
485 vkCmdSetScissor(command_buffer->handle, 0, 1, &scissor);
486
487 context.main_renderpass.w = context.framebuffer_width;
488 context.main_renderpass.h = context.framebuffer_height;
489
490 // Begin the render pass.
491 vulkan_renderpass_begin(
492 command_buffer,
493 &context.main_renderpass,
494 context.swapchain.framebuffers[context.image_index].handle);
495
496 return TRUE;
497}

499b8 vulkan_renderer_backend_end_frame(renderer_backend* backend, f32 delta_time) {
500 vulkan_command_buffer* command_buffer = &context.graphics_command_buffers[context.image_index];
501
502 // End renderpass
503 vulkan_renderpass_end(command_buffer, &context.main_renderpass);
504
505 vulkan_command_buffer_end(command_buffer);
506
507 // Make sure the previous frame is not using this image (i.e. its fence is being waited on)
508 if (context.images_in_flight[context.image_index] != VK_NULL_HANDLE) { // was frame
509 vulkan_fence_wait(
510 &context,
511 context.images_in_flight[context.image_index],
512 UINT64_MAX);
513 }
514
515 // Mark the image fence as in-use by this frame.
516 context.images_in_flight[context.image_index] = &context.in_flight_fences[context.current_frame];
517
518 // Reset the fence for use on the next frame
519 vulkan_fence_reset(&context, &context.in_flight_fences[context.current_frame]);
520
521 // Submit the queue and wait for the operation to complete.
522 // Begin queue submission
523 VkSubmitInfo submit_info = {VK_STRUCTURE_TYPE_SUBMIT_INFO};
524
525 // Command buffer(s) to be executed.
526 submit_info.commandBufferCount = 1;
527 submit_info.pCommandBuffers = &command_buffer->handle;
528
529 // The semaphore(s) to be signaled when the queue is complete.
530 submit_info.signalSemaphoreCount = 1;
531 submit_info.pSignalSemaphores = &context.queue_complete_semaphores[context.current_frame];
532
533 // Wait semaphore ensures that the operation cannot begin until the image is available.
534 submit_info.waitSemaphoreCount = 1;
535 submit_info.pWaitSemaphores = &context.image_available_semaphores[context.current_frame];
536
537 /*
538 * Each semaphore waits on the corresponding pipeline stage to complete. 1:1 ratio.
539 * VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT prevents subsequent colour attachment
540 * writes from executing until the semaphore signals (i.e. one frame is presented at a time)
541 */
542 VkPipelineStageFlags flags[1] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
543 submit_info.pWaitDstStageMask = flags;
544
545 VkResult result = vkQueueSubmit(
546 context.device.graphics_queue,
547 1,
548 &submit_info,
549 context.in_flight_fences[context.current_frame].handle);
550
551 if (result != VK_SUCCESS) {
552 KERROR("vkQueueSubmit failed with result: %s", vulkan_result_string(result, TRUE));
553
554 return FALSE;
555 }
556
557 vulkan_command_buffer_update_submitted(command_buffer); // End queue submission
558
559 // Give the image back to the swapchain.
560 vulkan_swapchain_present(
561 &context,
562 &context.swapchain,
563 context.device.graphics_queue,
564 context.device.present_queue,
565 context.queue_complete_semaphores[context.current_frame],
566 context.image_index);
567
568 return TRUE;
569}

594i32 find_memory_index(u32 type_filter, u32 property_flags) {
595 KDEBUG("Finding memory index for filter %u, flags %u...", type_filter, property_flags);
596
597 VkPhysicalDeviceMemoryProperties memory_properties;
598 vkGetPhysicalDeviceMemoryProperties(context.device.physical_device, &memory_properties);
599
600 for (u32 i = 0; i < memory_properties.memoryTypeCount; ++i) {
601 // Check each memory type to see if its bit is set to 1.
602 if (type_filter & (1 << i) && (memory_properties.memoryTypes[i].propertyFlags & property_flags) == property_flags) {
603 KDEBUG("Found valid memory index: %u", i);
604 return i;
605 }
606 }
607
608 KWARN("Unable to find suitable memory type!");
609 return -1;
610}

659b8 recreate_swapchain(renderer_backend* backend) {
660 KINFO("Recreating");
661 // If already being recreated, do not try again.
662 if (context.recreating_swapchain) {
663 KDEBUG("recreate_swapchain called when already recreating, skipping!");
664 return FALSE;
665 }
666
667 // Detect if the window is too small to be drawn to
668 if (context.framebuffer_width == 0 || context.framebuffer_height == 0) {
669 KDEBUG("recreate_swapchain called when window is < 1 in a dimension. Booting.");
670 return FALSE;
671 }
672
673 // Mark as recreating if the dimensions are valid.
674 context.recreating_swapchain = TRUE;
675
676 // Wait for any operations to complete.
677 vkDeviceWaitIdle(context.device.logical_device);
678
679 // Clear these out just in case.
680 for (u32 i = 0; i < context.swapchain.image_count; ++i) {
681 context.images_in_flight[i] = 0;
682 }
683
684 // Requery support
685 vulkan_device_query_swapchain_support(
686 context.device.physical_device,
687 context.surface,
688 &context.device.swapchain_support);
689 vulkan_device_detect_depth_format(&context.device);
690
691 vulkan_swapchain_recreate(
692 &context,
693 cached_framebuffer_width,
694 cached_framebuffer_height,
695 &context.swapchain);
696
697 // Sync the framebuffer size with the cached sizes.
698 context.framebuffer_width = cached_framebuffer_width;
699 context.framebuffer_height = cached_framebuffer_height;
700 context.main_renderpass.w = context.framebuffer_width;
701 context.main_renderpass.h = context.framebuffer_height;
702 cached_framebuffer_width = 0;
703 cached_framebuffer_height = 0;
704
705 // Update framebuffer size generation.
706 context.framebuffer_size_last_generation = context.framebuffer_size_generation;
707
708 // Cleanup swapchain
709 for (u32 i = 0; i < context.swapchain.image_count; ++i) {
710 vulkan_command_buffer_free(&context, context.device.graphics_command_pool, &context.graphics_command_buffers[i]);
711 }
712
713 // Framebuffers.
714 for (u32 i = 0; i < context.swapchain.image_count; ++i) {
715 vulkan_framebuffer_destroy(&context, &context.swapchain.framebuffers[i]);
716 }
717
718 context.main_renderpass.x = 0;
719 context.main_renderpass.y = 0;
720 context.main_renderpass.w = context.framebuffer_width;
721 context.main_renderpass.h = context.framebuffer_height;
722
723 regenerate_framebuffers(backend, &context.swapchain, &context.main_renderpass);
724
725 create_command_buffers(backend);
726
727 // Clear the recreating flag.
728 context.recreating_swapchain = FALSE;
729
730 return TRUE;

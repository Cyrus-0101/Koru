432i32 find_memory_index(u32 type_filter, u32 property_flags) {
433 KDEBUG("Finding memory index for filter %u, flags %u...", type_filter, property_flags);
434
435 VkPhysicalDeviceMemoryProperties memory_properties;
436 vkGetPhysicalDeviceMemoryProperties(context.device.physical_device, &memory_properties);
437
438 for (u32 i = 0; i < memory_properties.memoryTypeCount; ++i) {
439 // Check each memory type to see if its bit is set to 1.
440 if (type_filter & (1 << i) && (memory_properties.memoryTypes[i].propertyFlags & property_flags) == property_flags) {
441 KDEBUG("Found valid memory index: %u", i);
442 return i;
443 }
444 }
445
446 KWARN("Unable to find suitable memory type!");
447 return -1;
448}

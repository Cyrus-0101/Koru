159
160 // Don't care about the old layout - transition to optimal layout (for the underlying implementation).
161 if (old_layout == VK_IMAGE_LAYOUT_UNDEFINED && new_layout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
162 barrier.srcAccessMask = 0;
163
164 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
165
166 // Don't care what stage the pipeline is in at the start.
167 source_stage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
168
169 // Used for copying
170 dest_stage = VK_PIPELINE_STAGE_TRANSFER_BIT;
171 } else if (old_layout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL && new_layout == VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
172 // Transitioning from a transfer destination layout to a shader-readonly layout.
173 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
174 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
175
176 // From a copying stage to...
177 source_stage = VK_PIPELINE_STAGE_TRANSFER_BIT;
178
179 // The fragment stage.
180 dest_stage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
181 } else {
182 KFATAL("unsupported layout transition!");
183
184 return;
185 }

187 vkCmdPipelineBarrier(
188 command_buffer->handle,
189 source_stage, dest_stage,
190 0,
191 0, 0,
192 0, 0,
193 1, &barrier);

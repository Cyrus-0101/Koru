87b8 geometry_system_initialize(u64* memory_requirement, void* state, geometry_system_config config) {
88 if (config.max_geometry_count == 0) {
89 KFATAL("geometry_system_initialize - config.max_geometry_count must be > 0.");
90 return False;
91 }
92
93 // Block of memory will contain state structure, then block for array, then block for hashtable.
94 u64 struct_requirement = sizeof(geometry_system_state);
95 u64 array_requirement = sizeof(geometry) * config.max_geometry_count;
96 *memory_requirement = struct_requirement + array_requirement;
97
98 if (!state) {
99 return True;
100 }
101
102 state_ptr = state;
103 state_ptr->config = config;
104
105 // The array block is after the state. Already allocated, so just set the pointer.
106 void* array_block = state + struct_requirement;
107 state_ptr->registered_geometries = array_block;
108
109 // Invalidate all geometries in the array.
110 u32 count = state_ptr->config.max_geometry_count;
111 for (u32 i = 0; i < count; ++i) {
112 state_ptr->registered_geometries[i].geometry.id = INVALID_ID;
113 state_ptr->registered_geometries[i].geometry.internal_id = INVALID_ID;
114 state_ptr->registered_geometries[i].geometry.generation = INVALID_ID;
115 }
116
117 if (!create_default_geometry(state_ptr)) {
118 KFATAL("Failed to create default geometry. Application cannot continue.");
119 return False;
120 }
121
122 return True;
123}

140geometry* geometry_system_acquire_from_config(geometry_config config, b8 auto_release) {
141 geometry* g = 0;
142 for (u32 i = 0; i < state_ptr->config.max_geometry_count; ++i) {
143 if (state_ptr->registered_geometries[i].geometry.id == INVALID_ID) {
144 // Found empty slot.
145 state_ptr->registered_geometries[i].auto_release = auto_release;
146 state_ptr->registered_geometries[i].reference_count = 1;
147 g = &state_ptr->registered_geometries[i].geometry;
148 g->id = i;
149 break;
150 }
151 }
152

154 KERROR("Unable to obtain free slot for geometry. Adjust configuration to allow more space. Returning nullptr.");
155 return 0;
156 }
157

159 KERROR("Failed to create geometry. Returning nullptr.");
160 return 0;
161 }

166void geometry_system_release(geometry* geometry) {
167 if (geometry && geometry->id != INVALID_ID) {
168 geometry_reference* ref = &state_ptr->registered_geometries[geometry->id];
169
170 // Take a copy of the id;
171 u32 id = geometry->id;
172 if (ref->geometry.id == geometry->id) {
173 if (ref->reference_count > 0) {
174 ref->reference_count--;
175 }
176
177 // Also blanks out the geometry id.
178 if (ref->reference_count < 1 && ref->auto_release) {
179 destroy_geometry(state_ptr, &ref->geometry);
180 ref->reference_count = 0;
181 ref->auto_release = False;
182 }
183 } else {
184 KFATAL("Geometry id mismatch. Check registration logic, as this should never occur.");
185 }
186 return;
187 }
188
189 KWARN("geometry_system_acquire_by_id cannot release invalid geometry id. Nothing was done.");
190}

201b8 create_geometry(geometry_system_state* state, geometry_config config, geometry* g) {
202 // Send the geometry off to the renderer to be uploaded to the GPU.
203 if (!renderer_create_geometry(g, config.vertex_count, config.vertices, config.index_count, config.indices)) {
204 // Invalidate the entry.
205 state->registered_geometries[g->id].reference_count = 0;
206 state->registered_geometries[g->id].auto_release = False;
207 g->id = INVALID_ID;
208 g->generation = INVALID_ID;
209 g->internal_id = INVALID_ID;
210
211 return False;
212 }
213
214 // Acquire the material
215 if (string_length(config.material_name) > 0) {
216 g->material = material_system_acquire(config.material_name);
217 if (!g->material) {
218 g->material = material_system_get_default();
219 }
220 }
221
222 return True;
223}

236 g->material = 0;
